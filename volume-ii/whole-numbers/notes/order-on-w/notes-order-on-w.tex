\subsection*{The Extended Order}

\begin{definitionbox}{Definition (Order on \texorpdfstring{$\mathbb{W}$}{W})}
\begin{definition}[Order on $\mathbb{W}$]
\label{def:order-on-w}
Define $\leq_{\mathbb{W}}$ on $\mathbb{W}$ by declaring
\[
0\leq_{\mathbb{W}}a\quad(a\in\mathbb{W}),
\]
and, for $a,b\in\mathbb{N}$,
\[
a\leq_{\mathbb{W}}b
\quad\Longleftrightarrow\quad
a\leq_{\mathbb{N}}b.
\]
\end{definition}
\end{definitionbox}

\begin{definitionbox}{Definition (Strict Order on \texorpdfstring{$\mathbb{W}$}{W})}
\begin{definition}[Strict Order on $\mathbb{W}$]
\label{def:strict-order-on-w}
Define $<_{\mathbb{W}}$ on $\mathbb{W}$ by declaring
\[
0<_{\mathbb{W}}n\quad(n\in\mathbb{N}),
\]
and, for $a,b\in\mathbb{N}$,
\[
a<_{\mathbb{W}}b
\quad\Longleftrightarrow\quad
a<_{\mathbb{N}}b.
\]
No element of $\mathbb{W}$ is strictly less than $0$.
\end{definition}
\end{definitionbox}

\begin{remark*}[Interpretation]
The non-strict order makes $0$ least. The strict order says exactly that the
positive natural numbers lie after $0$ and keep their old strict ordering.
\end{remark*}

\begin{theorembox}{Theorem (Order on \texorpdfstring{$\mathbb{W}$}{W} Extends Order on \texorpdfstring{$\mathbb{N}$}{N})}
\begin{theorem}[Order on $\mathbb{W}$ Extends Order on $\mathbb{N}$]
\label{thm:order-on-w-extends-order-on-n}
For all $a,b\in\mathbb{N}$,
\[
a\leq_{\mathbb{W}}b\Longleftrightarrow a\leq_{\mathbb{N}}b,
\qquad
a<_{\mathbb{W}}b\Longleftrightarrow a<_{\mathbb{N}}b.
\]
\hyperref[prf:order-on-w-extends-order-on-n]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{theorembox}{Theorem (Order on \texorpdfstring{$\mathbb{W}$}{W} Is Antisymmetric)}
\begin{theorem}[Order on $\mathbb{W}$ Is Antisymmetric]
\label{thm:order-on-w-antisymmetric}
For all $a,b\in\mathbb{W}$,
\[
a\leq_{\mathbb{W}}b\ \text{and}\ b\leq_{\mathbb{W}}a
\Longrightarrow a=b.
\]
\hyperref[prf:order-on-w-antisymmetric]{Go to proof.}
\end{theorem}
