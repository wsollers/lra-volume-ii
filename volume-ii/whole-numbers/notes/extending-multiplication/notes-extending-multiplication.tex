\subsection*{The Extended Multiplication Operation}

\begin{definitionbox}{Definition (Multiplication on \texorpdfstring{$\mathbb{W}$}{W})}
\begin{definition}[Multiplication on $\mathbb{W}$]
\label{def:multiplication-on-w}
Define $\cdot_{\mathbb{W}}:\mathbb{W}\times\mathbb{W}\to\mathbb{W}$ by
\[
0\cdot_{\mathbb{W}}a=0,\qquad
a\cdot_{\mathbb{W}}0=0,
\]
and, for $a,b\in\mathbb{N}$,
\[
a\cdot_{\mathbb{W}}b=a\cdot_{\mathbb{N}}b.
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\cdot_{\mathbb{W}}:\mathbb{W}\times\mathbb{W}\to\mathbb{W},\\
&\forall a\in\mathbb{W}\;(0\cdot_{\mathbb{W}}a=0\ \wedge\ a\cdot_{\mathbb{W}}0=0),\\
&\forall a,b\in\mathbb{N}\;
  (a\cdot_{\mathbb{W}}b=a\cdot_{\mathbb{N}}b).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
The zero cases are new. The positive-positive case is inherited from
$\mathbb{N}$. Since $0\notin\mathbb{N}$, these clauses do not compete.
\end{remark*}

\begin{theorembox}{Theorem (Multiplication on \texorpdfstring{$\mathbb{W}$}{W} Is Well-Defined)}
\begin{theorem}[Multiplication on $\mathbb{W}$ Is Well-Defined]
\label{thm:multiplication-on-w-well-defined}
The case definition of $\cdot_{\mathbb{W}}$ determines a binary operation on
$\mathbb{W}$.
\hyperref[prf:multiplication-on-w-well-defined]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\cdot_{\mathbb{W}}\text{ is a well-defined binary operation on }\mathbb{W}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
The inherited positive case is already well-defined on $\mathbb{N}$, and the
new zero cases land in the distinguished element $0\in\mathbb{W}$.
\end{remark*}

\begin{theorembox}{Theorem (Zero Is Absorbing for Multiplication on \texorpdfstring{$\mathbb{W}$}{W})}
\begin{theorem}[Zero Is Absorbing for Multiplication on $\mathbb{W}$]
\label{thm:zero-absorbing-for-multiplication-on-w}
For every $a\in\mathbb{W}$,
\[
0\cdot_{\mathbb{W}}a=0
\qquad\text{and}\qquad
a\cdot_{\mathbb{W}}0=0.
\]
\hyperref[prf:zero-absorbing-for-multiplication-on-w]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\forall a\in\mathbb{W}\;
\bigl(0\cdot_{\mathbb{W}}a=0\ \wedge\ a\cdot_{\mathbb{W}}0=0\bigr).
\]
\end{remark*}

\begin{remark*}[Interpretation]
The adjoined element $0$ is absorbing for multiplication on both sides.
\end{remark*}

\begin{theorembox}{Theorem (One Is a Multiplicative Identity on \texorpdfstring{$\mathbb{W}$}{W})}
\begin{theorem}[One Is a Multiplicative Identity on $\mathbb{W}$]
\label{thm:one-multiplicative-identity-on-w}
For every $a\in\mathbb{W}$,
\[
1\cdot_{\mathbb{W}}a=a
\qquad\text{and}\qquad
a\cdot_{\mathbb{W}}1=a.
\]
\hyperref[prf:one-multiplicative-identity-on-w]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\forall a\in\mathbb{W}\;
\bigl(1\cdot_{\mathbb{W}}a=a\ \wedge\ a\cdot_{\mathbb{W}}1=a\bigr).
\]
\end{remark*}

\begin{remark*}[Interpretation]
The old multiplicative identity remains an identity after adjoining $0$.
\end{remark*}

\begin{theorembox}{Theorem (Multiplication on \texorpdfstring{$\mathbb{W}$}{W} Extends Multiplication on \texorpdfstring{$\mathbb{N}$}{N})}
\begin{theorem}[Multiplication on $\mathbb{W}$ Extends Multiplication on $\mathbb{N}$]
\label{thm:multiplication-on-w-extends-multiplication-on-n}
For all $a,b\in\mathbb{N}$,
\[
a\cdot_{\mathbb{W}}b=a\cdot_{\mathbb{N}}b.
\]
\hyperref[prf:multiplication-on-w-extends-multiplication-on-n]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\forall a,b\in\mathbb{N}\;
\bigl(a\cdot_{\mathbb{W}}b=a\cdot_{\mathbb{N}}b\bigr).
\]
\end{remark*}

\begin{remark*}[Interpretation]
The extension does not alter products of positive natural numbers.
\end{remark*}

\begin{theorembox}{Theorem (Multiplication on \texorpdfstring{$\mathbb{W}$}{W} Is Associative)}
\begin{theorem}[Multiplication on $\mathbb{W}$ Is Associative]
\label{thm:multiplication-on-w-associative}
For all $a,b,c\in\mathbb{W}$,
\[
(a\cdot_{\mathbb{W}}b)\cdot_{\mathbb{W}}c
=
a\cdot_{\mathbb{W}}(b\cdot_{\mathbb{W}}c).
\]
\hyperref[prf:multiplication-on-w-associative]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\forall a,b,c\in\mathbb{W}\;
\bigl((a\cdot_{\mathbb{W}}b)\cdot_{\mathbb{W}}c
=a\cdot_{\mathbb{W}}(b\cdot_{\mathbb{W}}c)\bigr).
\]
\end{remark*}

\begin{remark*}[Interpretation]
Associativity follows from the absorbing zero cases and the inherited
associativity of multiplication on positive natural numbers.
\end{remark*}

\begin{theorembox}{Theorem (Multiplication on \texorpdfstring{$\mathbb{W}$}{W} Is Commutative)}
\begin{theorem}[Multiplication on $\mathbb{W}$ Is Commutative]
\label{thm:multiplication-on-w-commutative}
For all $a,b\in\mathbb{W}$,
\[
a\cdot_{\mathbb{W}}b=b\cdot_{\mathbb{W}}a.
\]
\hyperref[prf:multiplication-on-w-commutative]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\forall a,b\in\mathbb{W}\;
\bigl(a\cdot_{\mathbb{W}}b=b\cdot_{\mathbb{W}}a\bigr).
\]
\end{remark*}

\begin{remark*}[Interpretation]
The zero cases are symmetric, and the positive-positive case is the inherited
commutativity theorem for $\mathbb{N}$.
\end{remark*}

\begin{theorembox}{Theorem (Multiplication Distributes over Addition on \texorpdfstring{$\mathbb{W}$}{W})}
\begin{theorem}[Multiplication Distributes over Addition on $\mathbb{W}$]
\label{thm:multiplication-distributes-over-addition-on-w}
For all $a,b,c\in\mathbb{W}$,
\[
a\cdot_{\mathbb{W}}(b+_{\mathbb{W}}c)
=
(a\cdot_{\mathbb{W}}b)+_{\mathbb{W}}(a\cdot_{\mathbb{W}}c).
\]
\hyperref[prf:multiplication-distributes-over-addition-on-w]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\forall a,b,c\in\mathbb{W}\;
\bigl(a\cdot_{\mathbb{W}}(b+_{\mathbb{W}}c)
=(a\cdot_{\mathbb{W}}b)+_{\mathbb{W}}(a\cdot_{\mathbb{W}}c)\bigr).
\]
\end{remark*}

\begin{remark*}[Interpretation]
Distributivity reduces zero cases to the absorbing law and positive cases to
the distributive law already proved on $\mathbb{N}$.
\end{remark*}
