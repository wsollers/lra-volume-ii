  title={\small\textbf{Theorem (\texorpdfstring{$\mathbb{W}$}{W} Has No Additive Inverses Except \texorpdfstring{$0$}{0})}},
  fonttitle=\small\bfseries]
\begin{theorem}[$\mathbb{W}$ Has No Additive Inverses Except $0$]
\label{thm:w-has-no-additive-inverses-except-zero}
If $a,b\in\mathbb{W}$ and
\[
a+_{\mathbb{W}}b=0,
\]
then $a=0$ and $b=0$.
\hyperref[prf:w-has-no-additive-inverses-except-zero]{Go to proof.}
\end{theorem}
\end{tcolorbox}

\begin{tcolorbox}[colback=thmbox, colframe=thmborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Theorem (\texorpdfstring{$\mathbb{W}$}{W} Is Not a Ring)}},
  fonttitle=\small\bfseries]
\begin{theorem}[$\mathbb{W}$ Is Not a Ring]
\label{thm:w-is-not-a-ring}
With the operations $+_{\mathbb{W}}$ and $\cdot_{\mathbb{W}}$, the whole
numbers do not form a ring.
\hyperref[prf:w-is-not-a-ring]{Go to proof.}
\end{theorem}
\end{tcolorbox}

\begin{remark*}[Interpretation]
The obstruction is not associativity, commutativity, or distributivity. The
obstruction is the absence of additive inverses for nonzero whole numbers.
\end{remark*}
