\begin{theorembox}{Theorem (\texorpdfstring{$(\mathbb{W},+,0)$}{(W,+,0)} Is a Commutative Monoid)}
\begin{theorem}[$(\mathbb{W},+,0)$ Is a Commutative Monoid]
\label{thm:w-additive-commutative-monoid}
The structure $(\mathbb{W},+_{\mathbb{W}},0)$ is a commutative monoid.
\hyperref[prf:w-additive-commutative-monoid]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
(\mathbb{W},+_{\mathbb{W}},0)\text{ is a commutative monoid}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
Addition on $\mathbb{W}$ has an identity element, is associative, and is
commutative.
\end{remark*}

\begin{theorembox}{Theorem (\texorpdfstring{$\mathbb{N}$}{N} Embeds into \texorpdfstring{$\mathbb{W}$}{W} as the Positive Part)}
\begin{theorem}[$\mathbb{N}$ Embeds into $\mathbb{W}$ as the Positive Part]
\label{thm:n-embeds-into-w-as-positive-part}
The inclusion map identifies $\mathbb{N}$ with the positive part
$\mathbb{W}\setminus\{0\}$ and preserves successor, $1$, addition,
multiplication, and order.
\hyperref[prf:n-embeds-into-w-as-positive-part]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\iota:\mathbb{N}\hookrightarrow\mathbb{W},\qquad \iota(n)=n,
\]
identifies $\mathbb{N}$ with $\mathbb{W}\setminus\{0\}$ and preserves
successor, $1$, addition, multiplication, and order.
\end{remark*}

\begin{remark*}[Interpretation]
The positive part of $\mathbb{W}$ is exactly the previously constructed
natural-number system, with the same arithmetic and order.
\end{remark*}

\begin{theorembox}{Theorem (\texorpdfstring{$\mathbb{W}$}{W} Has No Additive Inverses Except \texorpdfstring{$0$}{0})}
\begin{theorem}[$\mathbb{W}$ Has No Additive Inverses Except $0$]
\label{thm:w-has-no-additive-inverses-except-zero}
If $a,b\in\mathbb{W}$ and
\[
a+_{\mathbb{W}}b=0,
\]
then $a=0$ and $b=0$.
\hyperref[prf:w-has-no-additive-inverses-except-zero]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\forall a,b\in\mathbb{W}\;
\bigl(a+_{\mathbb{W}}b=0 \Longrightarrow (a=0\ \wedge\ b=0)\bigr).
\]
\end{remark*}

\begin{remark*}[Interpretation]
The only way to sum to zero inside $\mathbb{W}$ is for both summands already
to be zero.
\end{remark*}

\begin{theorembox}{Theorem (\texorpdfstring{$\mathbb{W}$}{W} Is Not a Ring)}
\begin{theorem}[$\mathbb{W}$ Is Not a Ring]
\label{thm:w-is-not-a-ring}
With the operations $+_{\mathbb{W}}$ and $\cdot_{\mathbb{W}}$, the whole
numbers do not form a ring.
\hyperref[prf:w-is-not-a-ring]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
(\mathbb{W},+_{\mathbb{W}},\cdot_{\mathbb{W}},0,1)
\text{ is not a ring}.
\]
\end{remark*}

\begin{remark*}[Interpretation]
The obstruction is not associativity, commutativity, or distributivity. The
obstruction is the absence of additive inverses for nonzero whole numbers.
\end{remark*}
