\begin{theorem*}[General Recursion Theorem for a Peano System]
Let $(P,S,1)$ be a Peano system, let $X$ be a set, let $a \in X$, and let
\[
\Phi : P \times X \to X
\]
be a function. Then there exists a unique function
\[
f : P \to X
\]
such that
\[
f(1)=a
\]
and
\[
\forall n \in P\;(f(S(n))=\Phi(n,f(n))).
\]
\end{theorem*}
