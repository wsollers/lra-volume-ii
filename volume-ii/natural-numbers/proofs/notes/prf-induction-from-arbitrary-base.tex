\begin{theorem*}[Induction from an Arbitrary Base]
Let $(P,S,1)$ be a Peano system, let $m \in P$, and let $\varphi$ be a
predicate on $P$. If $\varphi(m)$ and
$\forall n \in P\;(n \ge m \land \varphi(n) \Longrightarrow \varphi(S(n)))$,
then $\forall n \in P\;(n \ge m \Longrightarrow \varphi(n))$.
\end{theorem*}
