\section{Axioms for \texorpdfstring{$\mathbb{N}$}{N}}
\label{sec:nz-axioms}

\begin{topicbox}{The Canonical Peano System}
\begin{exposition}
The natural numbers are now fixed as the canonical one-based Peano system.
The preceding Peano chapter developed the abstract interface $(X,e,S)$; in
this chapter, $\mathbb{N}$ is the distinguished instance obtained by taking
the base element to be $1$. The symbol $0$ is not part of this system. It is
adjoined in the next chapter when the whole numbers are constructed.
\end{exposition}

\begin{definitionbox}{Definition (The Natural Numbers)}
\begin{definition}[The Natural Numbers]
\label{def:natural-numbers}
The natural numbers $\mathbb{N}$ are the canonical Peano system
\[
(\mathbb{N},1,S),
\]
where $1 \in \mathbb{N}$ is the distinguished base element and
$S:\mathbb{N}\to\mathbb{N}$ is the successor operation.
\end{definition}
\end{definitionbox}

\begin{remark*}[Standing convention]
Throughout this chapter, variables $m,n,k$ range over $\mathbb{N}$ unless a
larger ambient set is explicitly named. The element $1$ is the base natural
number, and $S(n)$ is the successor of $n$.
\end{remark*}

\begin{topicbox}{Peano Axioms Specialized to \texorpdfstring{$\mathbb{N}$}{N}}
\begin{exposition}
The following axioms are not new principles beyond the Peano-system interface;
they are the abstract Peano axioms specialized to the canonical natural-number
system. Naming them here makes later references inside the arithmetic chapter
point directly to $\mathbb{N}$ rather than to an arbitrary Peano system.
\end{exposition}

\begin{axiombox}{Axiom (N1, Base)}
\begin{axiom}[N1, Base]
\label{ax:n-base}
\[
1 \in \mathbb{N}.
\]
\end{axiom}
\end{axiombox}

\begin{axiombox}{Axiom (N2, Successor Closure)}
\begin{axiom}[N2, Successor Closure]
\label{ax:n-successor-closure}
\[
\forall n \in \mathbb{N}\;(S(n)\in \mathbb{N}).
\]
\end{axiom}
\end{axiombox}

\begin{axiombox}{Axiom (N3, Base Has No Predecessor)}
\begin{axiom}[N3, Base Has No Predecessor]
\label{ax:n-base-has-no-predecessor}
\[
\forall n \in \mathbb{N}\;(S(n)\ne 1).
\]
\end{axiom}

\begin{axiombox}{Axiom (N4, Successor Injectivity)}
\begin{axiom}[N4, Successor Injectivity]
\label{ax:n-successor-injective}
\[
\forall m,n \in \mathbb{N}\;(S(m)=S(n)\Longrightarrow m=n).
\]
\end{axiom}
\end{axiombox}

\begin{axiombox}{Axiom (N5, Induction)}
\begin{axiom}[N5, Induction]
\label{ax:n-induction}
For every predicate $\varphi$ on $\mathbb{N}$,
\[
\Bigl(\varphi(1)\land
\forall n\in\mathbb{N}\;(\varphi(n)\Longrightarrow \varphi(S(n)))\Bigr)
\Longrightarrow
\forall n\in\mathbb{N}\;\varphi(n).
\]
\end{axiom}
\end{axiombox}
