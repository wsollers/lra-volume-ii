\section{Recursion on \texorpdfstring{$\mathbb{N}$}{N}}

\begin{topicbox}{Iterator Data and the Peano Iterator Theorem}
\begin{exposition}
Recursion begins with the problem of generating a function from a starting
value and a rule for passing to the next stage. In a Peano system, the
successor operation provides the stage structure. Iterator data consists of a
target set, an initial value, and a self-map on that target set. The iterator
theorem states that such data determines a unique function on the Peano
system.
\end{exposition}

\begin{tcolorbox}[colback=defbox, colframe=defborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Definition (Iterator Data on a Peano System)}},
  fonttitle=\small\bfseries]
\begin{definition}[Iterator Data on a Peano System]
\label{def:iterator-data-on-peano-system}
Let $(P,S,1)$ be a Peano system, and let $X$ be a set. A pair $(a,T)$ is
called \emph{iterator data on $X$} exactly when $a \in X$ and
\[
T : X \to X
\]
is a function.
\end{definition}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let } (P,S,1) \text{ be a Peano system and let } X \text{ be a set.}\\
&(a,T) \text{ is iterator data on } X
\Longleftrightarrow
\bigl(a \in X \land T : X \to X\bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\begin{aligned}
&\text{Let } (P,S,1) \text{ be a Peano system and let } X \text{ be a set.}\\
&(a,T) \text{ is not iterator data on } X
\Longleftrightarrow
\bigl(a \notin X \lor T \text{ is not a function } X \to X\bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Failure modes]
Iterator data fails in exactly two ways: the proposed initial value may lie
outside the target set, or the proposed update rule may fail to be a self-map
of that target set.
\end{remark*}

\begin{remark*}[Failure mode decomposition]
\[
\begin{aligned}
&(a,T) \text{ is not iterator data on } X\\
\Longleftrightarrow\;&
\underbrace{a \notin X}_{\text{invalid initial value}}\\
&\lor
\underbrace{T \text{ is not a function } X \to X}_{\text{invalid update rule}}.
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
Iterator data specifies the target of the recursive construction, the first
value to be assigned, and the update rule that carries one stage to the next.
The Peano system supplies the indexing structure; the pair $(a,T)$ supplies the
data to be propagated along that structure.
\end{remark*}

\begin{remark*}[Comparison with Feferman]
The pair $(a,T)$ used here corresponds to the initial value $c$ and the update
function $G$ in Theorem 3.4 of \citet{FefermanNumberSystems1964}. The present
notes package this data into a reusable named structure called iterator data.
\end{remark*}

\begin{tcolorbox}[colback=thmbox, colframe=thmborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Theorem (Peano Iterator Theorem)}},
  fonttitle=\small\bfseries]
\begin{theorem}[Peano Iterator Theorem]
\label{thm:peano-iterator-theorem}
Let $(P,S,1)$ be a Peano system, let $X$ be a set, and let $(a,T)$ be iterator
data on $X$. Then there exists a unique function
\[
f : P \to X
\]
such that
\[
f(1)=a
\qquad\text{and}\qquad
\forall n \in P\;(f(S(n))=T(f(n))).
\]
\hyperref[prf:peano-iterator-theorem]{Go to proof.}
\end{theorem}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let } (P,S,1) \text{ be a Peano system, let } X \text{ be a set, and let } (a,T) \text{ be iterator data on } X.\\
&\exists! f : P \to X\;
\Bigl(f(1)=a \land
\forall n \in P\;(f(S(n))=T(f(n)))\Bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
The theorem states that a Peano system supports recursive definition by
iteration. Once the first value and the update rule are fixed, exactly one
function is generated. This is the basic mechanism from which recursively
defined arithmetic operations arise.
\end{remark*}

\begin{remark*}[Historical note]
This theorem corresponds directly to Theorem 3.4 in
\citet{FefermanNumberSystems1964}. It is the foundational recursion theorem
used to define arithmetic operations recursively on a Peano system.
\end{remark*}

\begin{topicbox}{Uniqueness and the Generated Iterator}
\begin{exposition}
The iterator theorem contains both existence and uniqueness. The next items
separate those consequences into reusable forms. The unique map determined by
iterator data is then given its own name so that later recursive
constructions can refer to it directly.
\end{exposition}

\begin{tcolorbox}[colback=corbox, colframe=corborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Corollary (Uniqueness of Recursively Defined Functions)}} ,
  fonttitle=\small\bfseries]
\begin{corollary}[Uniqueness of Recursively Defined Functions]
\label{cor:uniqueness-of-recursively-defined-functions}
Let $(P,S,1)$ be a Peano system, let $X$ be a set, and let $(a,T)$ be iterator
data on $X$. If $f,g : P \to X$ satisfy
\[
f(1)=a,\qquad g(1)=a,
\]
and
\[
\forall n \in P\;(f(S(n))=T(f(n)))
\qquad\text{and}\qquad
\forall n \in P\;(g(S(n))=T(g(n))),
\]
then
\[
f=g.
\]
\hyperref[prf:uniqueness-of-recursively-defined-functions]{Go to proof.}
\end{corollary}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let } (P,S,1) \text{ be a Peano system, let } X \text{ be a set, and let } (a,T) \text{ be iterator data on } X.\\
&\forall f,g : P \to X\;\Bigl(
f(1)=a \land g(1)=a \land
\forall n \in P\;(f(S(n))=T(f(n))) \\
&\qquad\qquad\qquad\qquad\land
\forall n \in P\;(g(S(n))=T(g(n)))
\Longrightarrow f=g\Bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
The recursive specification determines at most one function. Once the first
value and successor rule are fixed, no ambiguity remains in the values of the
resulting map.
\end{remark*}

\begin{remark*}[Comparison with Feferman]
Feferman proves existence and uniqueness together in Theorem 3.4 of
\citet{FefermanNumberSystems1964}. The present notes isolate the uniqueness
statement as a reusable corollary.
\end{remark*}

  \item \hyperref[thm:peano-iterator-theorem]{Peano Iterator Theorem}

\end{remark*}

\begin{tcolorbox}[colback=defbox, colframe=defborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Definition (Iterator-Generated Function)}},
  fonttitle=\small\bfseries]
\begin{definition}[Iterator-Generated Function]
\label{def:iterator-generated-function}
Let $(P,S,1)$ be a Peano system, let $X$ be a set, and let $(a,T)$ be iterator
data on $X$. The unique function
\[
f : P \to X
\]
satisfying
\[
f(1)=a
\qquad\text{and}\qquad
\forall n \in P\;(f(S(n))=T(f(n)))
\]
is called the \emph{iterator-generated function determined by $(a,T)$}.
\end{definition}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let } (P,S,1) \text{ be a Peano system, let } X \text{ be a set, and let } (a,T) \text{ be iterator data on } X.\\
&f \text{ is the iterator-generated function determined by } (a,T)\\
&\Longleftrightarrow
\Bigl(f : P \to X \land f(1)=a \land
\forall n \in P\;(f(S(n))=T(f(n)))\Bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\begin{aligned}
&\text{Let } (P,S,1) \text{ be a Peano system, let } X \text{ be a set, and let } (a,T) \text{ be iterator data on } X.\\
&f \text{ is not the iterator-generated function determined by } (a,T)\\
&\Longleftrightarrow
\Bigl(f : P \to X \land
\bigl(f(1)\neq a \lor \exists n \in P\;(f(S(n))\neq T(f(n)))\bigr)\Bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
This definition packages the unique output of the iterator theorem as a named
mathematical object. It turns the existence-and-uniqueness theorem into
notation that can be reused in later recursive constructions.
\end{remark*}

\begin{theorem}[Iterator Base Value]
\label{thm:iterator-base-value}
Let $(P,S,1)$ be a Peano system, let $X$ be a set, and let $(a,T)$ be iterator
data on $X$. If $f$ is the iterator-generated function determined by $(a,T)$,
then
\[
f(1)=a.
\]
\hyperref[prf:iterator-base-value]{Go to proof.}
\end{theorem}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let } (P,S,1) \text{ be a Peano system, let } X \text{ be a set, and let } (a,T) \text{ be iterator data on } X.\\
&f \text{ is the iterator-generated function determined by } (a,T)
\Longrightarrow f(1)=a.
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
The generated iterator begins at the prescribed initial value. The recursive
construction therefore starts exactly where the iterator data specifies.
\end{remark*}

\begin{topicbox}{Iteration and General Recursion}
\begin{exposition}
The iterator theorem first treats recursion driven by a fixed self-map on the
target set. That special case supplies the notion of iteration. More general
recursive definitions allow the successor step to depend on the current stage
as well as the current value.
\end{exposition}

\begin{tcolorbox}[colback=defbox, colframe=defborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Definition (Iteration of a Self-Map)}},
  fonttitle=\small\bfseries]
\begin{definition}[Iteration of a Self-Map]
\label{def:iteration-of-a-self-map}
Let $(P,S,1)$ be a Peano system, let $X$ be a set, let $a \in X$, and let
$T : X \to X$ be a function. The iterator-generated function determined by
$(a,T)$ is called the \emph{iteration of the self-map $T$ from the initial
value $a$ along $P$}.
\end{definition}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let } (P,S,1) \text{ be a Peano system, let } X \text{ be a set, let } a \in X, \text{ and let } T : X \to X.\\
&f \text{ is the iteration of } T \text{ from } a \text{ along } P\\
&\Longleftrightarrow
f \text{ is the iterator-generated function determined by } (a,T).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
Iteration packages recursive generation by repeated application of a single
self-map. This is the canonical recursion pattern behind repeated succession,
repeated addition, and repeated multiplication.
\end{remark*}

\begin{remark*}[Dependencies]
\begin{itemize}
  \item \hyperref[def:iterator-generated-function]{Iterator-Generated Function}
\end{itemize}
\end{remark*}

\begin{tcolorbox}[colback=defbox, colframe=defborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Definition (Recursive Function on a Peano System)}},
  fonttitle=\small\bfseries]
\begin{definition}[Recursive Function on a Peano System]
\label{def:recursive-function-on-peano-system}
Let $(P,S,1)$ be a Peano system, let $X$ be a set, let $a \in X$, and let
\[
\Phi : P \times X \to X
\]
be a function. A function
\[
f : P \to X
\]
is called \emph{recursive with initial value $a$ and step rule $\Phi$} exactly
when
\[
f(1)=a
\]
and
\[
\forall n \in P\;(f(S(n))=\Phi(n,f(n))).
\]
\end{definition}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let } (P,S,1) \text{ be a Peano system, let } X \text{ be a set, let } a \in X, \text{ and let } \Phi : P \times X \to X.\\
&f \text{ is recursive with initial value } a \text{ and step rule } \Phi\\
&\Longleftrightarrow
\Bigl(f : P \to X \land f(1)=a \land
\forall n \in P\;(f(S(n))=\Phi(n,f(n)))\Bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\begin{aligned}
&\text{Let } (P,S,1) \text{ be a Peano system, let } X \text{ be a set, let } a \in X, \text{ and let } \Phi : P \times X \to X.\\
&f \text{ is not recursive with initial value } a \text{ and step rule } \Phi\\
&\Longleftrightarrow
\Bigl(f : P \to X \land
\bigl(f(1)\neq a \lor \exists n \in P\;(f(S(n))\neq \Phi(n,f(n)))\bigr)\Bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Failure modes]
A function fails to satisfy the recursive specification if the initial value is
wrong or if the successor step fails at some stage of the Peano system.
\end{remark*}

\begin{remark*}[Failure mode decomposition]
\[
\begin{aligned}
&f \text{ is not recursive with initial value } a \text{ and step rule } \Phi\\
\Longleftrightarrow\;&
\underbrace{f(1)\neq a}_{\text{base clause fails}}\\
&\lor
\underbrace{\exists n \in P\;(f(S(n))\neq \Phi(n,f(n)))}_{\text{recursive step fails}}.
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
This definition allows the recursive step to depend on the current stage as
well as the current value. It generalizes pure iteration and captures the
standard form of recursive definition used throughout arithmetic.
\end{remark*}

\begin{remark*}[Comparison with Feferman]
This definition corresponds to the generalized recursive specification
described in Theorem 3.4' of \citet{FefermanNumberSystems1964}, where the
successor step is allowed to depend explicitly on both the stage and the
current value.
\end{remark*}

\begin{itemize}

\end{itemize}
\end{remark*}

\begin{tcolorbox}[colback=thmbox, colframe=thmborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Theorem (Recursive Specification Is Well-Defined)}},
  fonttitle=\small\bfseries]
\begin{theorem}[Recursive Specification Is Well-Defined]
\label{thm:recursive-specification-well-defined}
Let $(P,S,1)$ be a Peano system, let $X$ be a set, let $a \in X$, and let
\[
\Phi : P \times X \to X
\]
be a function. Then there exists at most one recursive function on $P$ with
initial value $a$ and step rule $\Phi$.
\hyperref[prf:recursive-specification-well-defined]{Go to proof.}
\end{theorem}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let } (P,S,1) \text{ be a Peano system, let } X \text{ be a set, let } a \in X, \text{ and let } \Phi : P \times X \to X.\\
&\forall f,g : P \to X\;\Bigl(
f(1)=a \land g(1)=a \land
\forall n \in P\;(f(S(n))=\Phi(n,f(n))) \\
&\qquad\qquad\qquad\qquad\land
\forall n \in P\;(g(S(n))=\Phi(n,g(n)))
\Longrightarrow f=g\Bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
Once the initial value and stage-dependent step rule are fixed, the recursive
specification cannot determine two different functions. This is the uniqueness
half of general recursion.
\end{remark*}

f(1)=a
\]

\begin{tcolorbox}[colback=thmbox, colframe=thmborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Theorem (Uniqueness of Peano Systems up to Isomorphism)}} ,
  fonttitle=\small\bfseries]
\begin{theorem}[Uniqueness of Peano Systems up to Isomorphism]
\label{thm:uniqueness-of-peano-systems-up-to-isomorphism}
Let $(P,S_P,1_P)$ and $(Q,S_Q,1_Q)$ be Peano systems. Then there exists a
unique bijection
\[
\theta : P \to Q
\]
such that
\[
\theta(1_P)=1_Q
\]
and
\[
\forall n \in P\;(\theta(S_P(n))=S_Q(\theta(n))).
\]
Thus any two Peano systems are isomorphic.
\hyperref[prf:uniqueness-of-peano-systems-up-to-isomorphism]{Go to proof.}
\end{theorem}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let } (P,S_P,1_P) \text{ and } (Q,S_Q,1_Q) \text{ be Peano systems.}\\
&\exists! \theta : P \to Q\;
\Bigl(\theta \text{ is bijective}
\land \theta(1_P)=1_Q\\
&\qquad\qquad\qquad\land
\forall n \in P\;(\theta(S_P(n))=S_Q(\theta(n)))\Bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
A Peano system has no extra structure beyond a distinguished first element and
a successor operation satisfying the Peano axioms. The theorem says that any
two such systems differ only by the names of their elements: there is exactly
one structure-preserving translation from one to the other.
\end{remark*}

\begin{remark*}[Historical note]
This theorem corresponds to Theorem 3.5 in
\citet{FefermanNumberSystems1964}. It records the categoricity of Peano
systems: once the Peano axioms and recursion theorem are in place, all Peano
systems are isomorphic.
\end{remark*}

The natural numbers are a chosen representative of the unique Peano-system

up to isomorphism, and only then fix a representative as the natural numbers.
The present notes keep the chosen representative explicit so that later
arithmetic constructions can be read either abstractly on a Peano system or
concretely on $\mathbb{N}$.

\begin{remark*}[Specialization to the natural numbers]
To recover the usual natural-number presentation, take $P=\mathbb{N}$, take
$S$ to be the successor function on $\mathbb{N}$, and take $1$ as the
distinguished first element. With that specialization, the abstract Peano
system notation becomes the familiar arithmetic setting used in the rest of
the chapter.
\end{remark*}
