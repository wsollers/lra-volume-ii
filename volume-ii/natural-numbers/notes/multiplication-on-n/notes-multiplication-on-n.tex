\section{Multiplication on \texorpdfstring{$\mathbb{N}$}{N}}

\begin{topicbox}{Recursive Definition of Multiplication}
\begin{exposition}
Multiplication on a Peano system is defined from addition by recursion. For
each fixed left input $m$, the map $n \mapsto m \cdot n$ is generated by
starting at $m$ and adding one further copy of $m$ at each successor stage of
the right input. This makes multiplication repeated addition rather than an
independent primitive operation.
\end{exposition}

\begin{tcolorbox}[colback=defbox, colframe=defborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Definition (Multiplication on a Peano System)}},
  fonttitle=\small\bfseries]
\begin{definition}[Multiplication on a Peano System]
\label{def:multiplication-on-peano-system}
Let $(P,S,1)$ be a Peano system. For each $m \in P$, define the function
\[
g_m : P \to P
\]
to be the unique function satisfying
\[
g_m(1)=m
\qquad\text{and}\qquad
\forall n \in P\;(g_m(S(n))=g_m(n)+m).
\]
For $m,n \in P$, define
\[
m \cdot n \coloneqq g_m(n).
\]
\end{definition}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let } (P,S,1) \text{ be a Peano system. For each } m \in P,
\text{ let } g_m : P \to P \text{ satisfy}\\
&g_m(1)=m
\qquad\text{and}\qquad
\forall n \in P\;(g_m(S(n))=g_m(n)+m).\\
&\text{Then for all } m,n \in P,\quad m \cdot n \coloneqq g_m(n).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Negated quantified statement]
\[
\begin{aligned}
&\text{Let } (P,S,1) \text{ be a Peano system. For each } m \in P,
\text{ let } g_m : P \to P \text{ satisfy}\\
&g_m(1)=m
\qquad\text{and}\qquad
\forall n \in P\;(g_m(S(n))=g_m(n)+m).\\
&\text{Then for all } m,n,x \in P,\quad x \neq m \cdot n
\Longleftrightarrow
x \neq g_m(n).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Failure modes]
The definition fails only when a proposed value for $m \cdot n$ does not agree
with the recursively generated function attached to the fixed left factor $m$.
Once $g_m$ is determined, every incorrect candidate product is simply a value
different from $g_m(n)$.
\end{remark*}

\begin{remark*}[Failure mode decomposition]
\[
\begin{aligned}
&\text{Let } (P,S,1) \text{ be a Peano system. For each } m \in P,
\text{ let } g_m : P \to P \text{ satisfy}\\
&g_m(1)=m
\qquad\text{and}\qquad
\forall n \in P\;(g_m(S(n))=g_m(n)+m).\\
&x \neq m \cdot n
\Longleftrightarrow
\underbrace{x \neq g_m(n)}_{\text{candidate value disagrees with the recursive output}}.
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
Multiplication is introduced by fixing the left input and recursively
generating the right-input map. The value $m \cdot 1$ is initialized at $m$,
and each successor step in the right input adds one further copy of $m$.
Multiplication is therefore repeated addition.
\end{remark*}

\begin{tcolorbox}[colback=thmbox, colframe=thmborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Theorem (Multiplication Is Well-Defined on a Peano System)}},
  fonttitle=\small\bfseries]
\begin{theorem}[Multiplication Is Well-Defined on a Peano System]
\label{thm:multiplication-well-defined-on-peano-system}
Let $(P,S,1)$ be a Peano system. For every $m \in P$, there exists a unique
function
\[
g_m : P \to P
\]
such that
\[
g_m(1)=m
\qquad\text{and}\qquad
\forall n \in P\;(g_m(S(n))=g_m(n)+m).
\]
Consequently the rule
\[
(m,n) \longmapsto m \cdot n \coloneqq g_m(n)
\]
defines a unique binary operation on $P$.
\hyperref[prf:multiplication-well-defined-on-peano-system]{Go to proof.}
\end{theorem}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let } (P,S,1) \text{ be a Peano system.}\\
&\forall m \in P\;\exists! g_m : P \to P\;
\Bigl(g_m(1)=m \land
\forall n \in P\;(g_m(S(n))=g_m(n)+m)\Bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
The recursive clauses for multiplication determine exactly one right-input map
for each fixed left input. The multiplication operation is therefore
constructed from recursion and addition rather than assumed at the outset.
\end{remark*}

\begin{topicbox}{Recursive Clauses as Reusable Equalities}
\begin{exposition}
The recursive definition of multiplication is given by a base value and a
successor step. As with addition, these clauses become more useful once they
are isolated as named theorems that can be cited directly in later algebraic
arguments.
\end{exposition}

\begin{tcolorbox}[colback=thmbox, colframe=thmborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Theorem (Multiplication With \texorpdfstring{$1$}{1})}},
  fonttitle=\small\bfseries]
\begin{theorem}[Multiplication With $1$]
\label{thm:multiplication-with-one}
Let $(P,S,1)$ be a Peano system. For every $m \in P$,
\[
m \cdot 1 = m.
\]
\hyperref[prf:multiplication-with-one]{Go to proof.}
\end{theorem}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let } (P,S,1) \text{ be a Peano system.}\\
&\forall m \in P\;(m \cdot 1 = m).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
This theorem records the base clause of the recursive definition in operation
notation. Multiplying by the distinguished first element leaves the left input
unchanged.
\end{remark*}

\begin{theorem}[Multiplication Successor on the Right]
\label{thm:multiplication-successor-on-right}
Let $(P,S,1)$ be a Peano system. For every $m,n \in P$,
\[
m \cdot S(n) = (m \cdot n) + m.
\]
\hyperref[prf:multiplication-successor-on-right]{Go to proof.}
\end{theorem}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let } (P,S,1) \text{ be a Peano system.}\\
&\forall m,n \in P\;(m \cdot S(n) = (m \cdot n) + m).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
Moving one successor step forward in the right input adds one further copy of
$m$ to the product. This is the recursive step of multiplication written in
operation notation.
\end{remark*}

\begin{topicbox}{Left Identity}
\begin{exposition}
The recursive definition controls multiplication when the right input evolves.
To use multiplication symmetrically, the first structural task is to
understand what happens when the distinguished first element appears on the
left. The next theorem identifies that left action explicitly.
\end{exposition}

\begin{tcolorbox}[colback=thmbox, colframe=thmborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Theorem (One Times \texorpdfstring{$n$}{n})}},
  fonttitle=\small\bfseries]
\begin{theorem}[One Times $n$]
\label{thm:one-times-n}
Let $(P,S,1)$ be a Peano system. For every $n \in P$,
\[
1 \cdot n = n.
\]
\hyperref[prf:one-times-n]{Go to proof.}
\end{theorem}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let } (P,S,1) \text{ be a Peano system.}\\
&\forall n \in P\;(1 \cdot n = n).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
The distinguished first element acts as a multiplicative identity on the left.
This is the multiplicative analogue of the theorem that $1+n$ produces the
successor of $n$ in the additive theory.
\end{remark*}

\begin{topicbox}{Distributivity and Associativity}
\begin{exposition}
Multiplication is built from repeated addition, so its first major structural
law is distributivity over addition. Once distributivity is available, the
operation can be shown to satisfy associativity, which makes iterated products
well-defined without ambiguity of grouping.
\end{exposition}

\begin{tcolorbox}[colback=thmbox, colframe=thmborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Theorem (Multiplication Distributes Over Addition)}},
  fonttitle=\small\bfseries]
\begin{theorem}[Multiplication Distributes Over Addition]
\label{thm:multiplication-distributes-over-addition}
Let $(P,S,1)$ be a Peano system. For every $a,b,c \in P$,
\[
a \cdot (b+c) = (a \cdot b) + (a \cdot c).
\]
\hyperref[prf:multiplication-distributes-over-addition]{Go to proof.}
\end{theorem}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let } (P,S,1) \text{ be a Peano system.}\\
&\forall a,b,c \in P\;(a \cdot (b+c) = (a \cdot b) + (a \cdot c)).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
Distributivity expresses formally that multiplying by $a$ means repeatedly
adding copies of $a$, so a sum in the second factor splits into two separate
blocks of repeated addition. This is the bridge between the additive and
multiplicative structures.
\end{remark*}

\begin{tcolorbox}[colback=thmbox, colframe=thmborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Theorem (Left Distributivity of Multiplication Over Addition)}},
  fonttitle=\small\bfseries]
\begin{theorem}[Left Distributivity of Multiplication Over Addition]
\label{thm:left-distributivity-of-multiplication-over-addition}
Let $(P,S,1)$ be a Peano system. For every $a,b,c \in P$,
\[
(a+b)\cdot c=(a\cdot c)+(b\cdot c).
\]
\hyperref[prf:left-distributivity-of-multiplication-over-addition]{Go to proof.}
\end{theorem}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let } (P,S,1) \text{ be a Peano system.}\\
&\forall a,b,c \in P\;\bigl((a+b)\cdot c=(a\cdot c)+(b\cdot c)\bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
The existing distributive law expands a sum in the right factor. This theorem
expands a sum in the left factor. It follows from right distributivity together
with commutativity of multiplication.
\end{remark*}

\begin{remark*}[Source alignment]
This is the house-style form of Feferman's left distributive law for
multiplication \citep{FefermanNumberSystems1964}.
\end{remark*}

  title={\small\textbf{Theorem (Multiplication Is Associative)}},
  fonttitle=\small\bfseries]
\begin{theorem}[Multiplication Is Associative]
\label{thm:multiplication-associative}
Let $(P,S,1)$ be a Peano system. For every $a,b,c \in P$,
\[
(a \cdot b) \cdot c = a \cdot (b \cdot c).
\]
\hyperref[prf:multiplication-associative]{Go to proof.}
\end{theorem}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let } (P,S,1) \text{ be a Peano system.}\\
&\forall a,b,c \in P\;((a \cdot b) \cdot c = a \cdot (b \cdot c)).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
Associativity shows that repeated multiplication can be regrouped without
changing the value. Once this law is established, longer products can be used
without parenthetical ambiguity.
\end{remark*}

\begin{topicbox}{Commutativity}
\begin{exposition}
The recursive definition of multiplication treats the right input as the
evolving variable, but the resulting operation is not meant to privilege one
factor over the other. The next theorem shows that the asymmetry of the
construction disappears in the final operation.
\end{exposition}

\begin{tcolorbox}[colback=thmbox, colframe=thmborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Theorem (Multiplication Is Commutative)}},
  fonttitle=\small\bfseries]
\begin{theorem}[Multiplication Is Commutative]
\label{thm:multiplication-commutative}
Let $(P,S,1)$ be a Peano system. For every $a,b \in P$,
\[
a \cdot b = b \cdot a.
\]
\hyperref[prf:multiplication-commutative]{Go to proof.}
\end{theorem}
\end{tcolorbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&\text{Let } (P,S,1) \text{ be a Peano system.}\\
&\forall a,b \in P\;(a \cdot b = b \cdot a).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
Although multiplication is defined recursively in one argument, the resulting
operation is symmetric in the two factors. Commutativity shows that repeated
addition of $a$ to build $a \cdot b$ agrees exactly with repeated addition of
$b$ to build $b \cdot a$.
\end{remark*}
