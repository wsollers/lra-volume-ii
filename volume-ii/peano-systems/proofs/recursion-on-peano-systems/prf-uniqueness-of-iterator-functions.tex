\begin{theorem*}[Uniqueness of Iterator Functions]
Let $(P,S,1)$ be a Peano system, and let $(W,c,g)$ be iterator data for it. If
$f,h:P\to W$ satisfy
\[
f(1)=c,\qquad h(1)=c,
\]
and
\[
\forall n\in P\;(f(S(n))=g(f(n)))
\qquad\text{and}\qquad
\forall n\in P\;(h(S(n))=g(h(n))),
\]
then $f=h$.
\end{theorem*}
