\begin{corollary*}[Predecessor Existence and Uniqueness Away from $1$]
Let $(P,S,1)$ be a Peano system, and let $x \in P$ with $x \neq 1$. Then
there exists a unique $u \in P$ such that $S(u)=x$.
\end{corollary*}

\begin{proof}
\textbf{Professional Standard Proof.}~\\
By
\hyperref[thm:every-element-is-one-or-a-successor]{Every Element Is Either
$1$ or a Successor}, either $x=1$ or there exists $u\in P$ such that
$S(u)=x$. Since $x\neq 1$, the first alternative is impossible. Hence there
exists $u\in P$ such that $S(u)=x$.

To prove uniqueness, let $v\in P$ and suppose $S(v)=x$. Since also $S(u)=x$,
we have
\[
S(v)=S(u).
\]
By injectivity of the successor map, $v=u$. Therefore there exists a unique
$u\in P$ such that $S(u)=x$.
\end{proof}

\begin{proof}
\textbf{Detailed Learning Proof.}~\\
We first prove existence. Since $x\in P$, the dichotomy theorem applies to
$x$. Thus either
\[
x=1
\]
or there exists some $u\in P$ such that
\[
S(u)=x.
\]
The hypothesis says $x\neq 1$, so the first case cannot occur. Therefore the
second case must occur, and a predecessor $u\in P$ of $x$ exists.

Now we prove uniqueness. Fix one such $u\in P$ with $S(u)=x$. Let
$v\in P$ be any other element satisfying
\[
S(v)=x.
\]
Then
\[
S(v)=x=S(u),
\]
so
\[
S(v)=S(u).
\]
The Peano successor map is injective, so $v=u$. Thus every predecessor of
$x$ is equal to the one already found. Hence the predecessor exists and is
unique.
\end{proof}

\begin{remark*}[Proof structure]
Existence comes from the dichotomy theorem after ruling out the case $x=1$.
Uniqueness is a direct application of injectivity of the successor map: two
elements with the same successor must be equal.
\end{remark*}
