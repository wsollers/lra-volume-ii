\begin{topicbox}{The Peano Axioms}
\begin{exposition}
The Peano axioms make precise what it means for a system to behave like
the natural numbers. They isolate the first element, the successor map,
and the induction principle. These axioms do not define addition,
multiplication, or order. They fix the structure that supports those
later constructions. The formulation used here belongs to the
Dedekind--Peano line of axiomatizing the natural-number structure
\citep{dedekind1901essays,PeanoArithmeticesPrincipia1889}.
\end{exposition}

\begin{definitionbox}{Definition (Peano System)}
\begin{definition}[Peano System]
\label{def:peano-system}
A triple $(P,S,1)$ is called a \emph{Peano system} exactly when $P$ is a set,
$1 \in P$, $S : P \to P$ is a function, and the following axioms hold:
\begin{enumerate}
  \item[\textbf{P1.}] $1 \in P$.
  \item[\textbf{P2.}] For every $n \in P$, $S(n) \in P$.
  \item[\textbf{P3.}] For every $n \in P$, $S(n) \neq 1$.
  \item[\textbf{P4.}] For all $m,n \in P$, if $S(m)=S(n)$ then $m=n$.
  \item[\textbf{P5.}] If $A \subseteq P$, $1 \in A$, and
  \[
    \forall n \in P\;(n \in A \Longrightarrow S(n) \in A),
  \]
  then $A=P$.
\end{enumerate}
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
\begin{aligned}
&(P,S,1) \text{ is a Peano system}\\
&\Longleftrightarrow
\Bigl(P \text{ is a set} \land 1 \in P \land S : P \to P \\
&\qquad \land \forall n \in P\;(S(n) \in P) \\
&\qquad \land \forall n \in P\;(S(n) \neq 1) \\
&\qquad \land \forall m,n \in P\;(S(m)=S(n) \Longrightarrow m=n) \\
&\qquad \land \forall A \subseteq P\;
\bigl((1 \in A \land
\forall n \in P\;(n \in A \Longrightarrow S(n) \in A))
\Longrightarrow A=P\bigr)\Bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Failure modes]
A triple $(P,S,1)$ fails to be a Peano system if the distinguished element
does not lie in the underlying set, if the successor map leaves the set, if
the successor hits $1$, if the successor identifies two distinct elements, or
if there exists a proper subset of $P$ that contains $1$ and is closed under
successor.
\end{remark*}

\begin{remark*}[Failure mode decomposition]
\[
\begin{aligned}
\neg\,\text{$(P,S,1)$ is a Peano system}
\Longrightarrow\;&
1 \notin P \\
&\lor \exists n \in P\;(S(n) \notin P) \\
&\lor \exists n \in P\;(S(n)=1) \\
&\lor \exists m,n \in P\;(m \neq n \land S(m)=S(n)) \\
&\lor \exists A \subseteq P\;
\bigl(1 \in A \land
\forall n \in P\;(n \in A \Longrightarrow S(n) \in A)
\land A \neq P\bigr).
\end{aligned}
\]
\end{remark*}

\begin{remark*}[Interpretation]
A Peano system is the abstract data of counting: an underlying set, a first
element, and a successor operation satisfying the usual non-collision and
induction principles. The purpose of the definition is to isolate the ambient
object before the individual axioms are recorded and used one by one.
\end{remark*}

\begin{remark*}[Source alignment]
The five displayed clauses are the house-style Peano-system presentation of
Peano's axiomatization of arithmetic \citep{PeanoArithmeticesPrincipia1889}.
They are stated here in structural form so that later definitions and proofs
can refer directly to the successor and induction clauses.
\end{remark*}

\begin{remark*}[Comparison with Feferman]
This definition corresponds closely to Definition 3.1 in
\citet{FefermanNumberSystems1964}, where Peano systems are introduced as the
abstract structural framework underlying the positive integers. The present
notes expand the definition into explicit quantified, failure-mode, and
decomposition forms in order to support later logical analysis and dependency
extraction.
\end{remark*}

\begin{axiombox}{Axiom (Base Element Lies in the Underlying Set)}
\begin{axiom}[Distinguished Element Lies in the Underlying Set]
\label{ax:peano-base-in-set}
\[
1 \in P.
\]
\end{axiom}
\end{axiombox}

\begin{remark*}[Standard quantified statement]
\[
1 \in P.
\]
\end{remark*}

\begin{remark*}[Interpretation]
The counting system begins from a distinguished first element of the
underlying set. This is the base point from which the remaining elements are
generated by the successor operation.
\end{remark*}

\begin{axiombox}{Axiom (Closure Under Successor)}
\begin{axiom}[Closure Under Successor]
\label{ax:peano-successor-closure}
If $n \in P$, then $S(n) \in P$.
\end{axiom}
\end{axiombox}

\begin{remark*}[Standard quantified statement]
\[
\forall n \in P\;(S(n) \in P).
\]
\end{remark*}

\begin{remark*}[Interpretation]
Applying the successor operation to an element of the underlying set never
leaves that set. Successor is therefore an internal operation on the counting
system.
\end{remark*}

\NoLocalDependencies

\begin{axiombox}{Axiom (Base Element Is Not a Successor)}
\begin{axiom}[Distinguished Element Is Not a Successor]
\label{ax:peano-base-not-successor}
For every $n \in P$, $S(n) \neq 1$.
\end{axiom}
\end{axiombox}

\begin{remark*}[Standard quantified statement]
\[
\forall n \in P\;(S(n) \neq 1).
\]
\end{remark*}

\begin{remark*}[Interpretation]
The first natural number does not arise by taking the successor of an earlier
natural number. This keeps the counting system from folding backward onto its
starting point.
\end{remark*}

\NoLocalDependencies

\begin{axiombox}{Axiom (Injectivity of Successor)}
\begin{axiom}[Injectivity of Successor]
\label{ax:peano-successor-injective}
If $m,n \in P$ and $S(m)=S(n)$, then $m=n$.
\end{axiom}
\end{axiombox}

\begin{remark*}[Standard quantified statement]
\[
\forall m,n \in P\;(S(m)=S(n) \Longrightarrow m=n).
\]
\end{remark*}

\begin{remark*}[Interpretation]
Different natural numbers do not collapse to the same successor. The successor
operation preserves distinctness and lets predecessor reasoning work whenever a
successor is known.
\end{remark*}

\NoLocalDependencies

\begin{axiombox}{Axiom (Induction Axiom)}
\begin{axiom}[Induction Axiom]
\label{ax:peano-induction}
Let $A \subseteq P$. If $1 \in A$ and if
\[
\forall n \in P\;(n \in A \Longrightarrow S(n) \in A),
\]
then $A=P$.
\end{axiom}
\end{axiombox}

\begin{remark*}[Standard quantified statement]
\[
\forall A \subseteq P\;
\Bigl(\bigl(1 \in A \land
\forall n \in P\;(n \in A \Longrightarrow S(n) \in A)\bigr)
\Longrightarrow A=P\Bigr).
\]
\end{remark*}

\begin{remark*}[Interpretation]
Any subset of the underlying set that contains the distinguished element and is
closed under successor must already contain the entire set. This axiom rules
out rogue elements and is the logical source of mathematical induction.
\end{remark*}

\NoLocalDependencies

\begin{axiombox}{Axiom (Existence of a Peano System)}
\begin{axiom}[Existence of a Peano System]
\label{ax:peano-system-existence}
There exists a Peano system.
\end{axiom}
\end{axiombox}

\begin{remark*}[Standard quantified statement]
\[
\exists P\;\exists S\;\bigl((P,S,1)\text{ is a Peano system}\bigr).
\]
\end{remark*}

\begin{remark*}[Interpretation]
The Peano axioms are not merely a formal pattern; at least one structure
satisfies them. This axiom guarantees that the abstract counting system is
available before we give it the conventional name $\mathbb{N}$.
\end{remark*}

\begin{remark*}[Historical note]
This axiom corresponds to Axiom 3.2 of
\citet{FefermanNumberSystems1964}, where the existence of at least one Peano
system is taken as a foundational assumption.
\end{remark*}
