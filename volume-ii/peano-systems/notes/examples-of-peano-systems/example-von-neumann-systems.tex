\subsection{Von Neumann Ordinals}

\begin{topicbox}{Set-Theoretic Carriers}
\begin{exposition}
Von Neumann ordinals realize natural-number structure inside set theory. In
this representation, each stage is a set, and successor is formed by adjoining
the current stage as a new element.
\end{exposition}

\begin{example*}[The Zero-Based Von Neumann Chain]
Let
\[
0=\varnothing,\qquad 1=\{0\},\qquad 2=\{0,1\},\qquad
3=\{0,1,2\},\quad \ldots
\]
Let $P=\omega$, let $e=0$, and define
\[
S:P\to P,
\qquad
S(n)=n\cup\{n\}.
\]
Then $(P,S,e)=(\omega,S,0)$ is a zero-based Peano system.
\end{example*}

\begin{remark*}[Structure]
The Peano data are
\[
\bigl(P,S,e\bigr)
=
\bigl(\omega,\,n\mapsto n\cup\{n\},\,0\bigr).
\]
This example uses the zero-based convention common in set-theoretic
foundations. The house convention for arithmetic in this volume remains
one-based.
\end{remark*}

\begin{example*}[The One-Based Von Neumann Chain]
Let
\[
P=\omega\setminus\{0\}, \qquad e=1,
\]
and define
\[
S:P\to P,
\qquad
S(n)=n\cup\{n\}.
\]
Then $(P,S,e)$ is a one-based Peano system.
\end{example*}

\begin{remark*}[Structure]
The Peano data are
\[
\bigl(P,S,e\bigr)
=
\bigl(\omega\setminus\{0\},\,n\mapsto n\cup\{n\},\,1\bigr).
\]
\end{remark*}

\begin{remark*}[Interpretation]
In the Von Neumann model, each natural number is literally the set of all
smaller natural numbers. This is not merely notation; it is a concrete
set-theoretic realization of the abstract Peano structure.
\end{remark*}
\end{topicbox}
