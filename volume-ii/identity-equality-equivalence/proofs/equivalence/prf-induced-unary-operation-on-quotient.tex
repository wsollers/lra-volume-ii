\begin{theorem*}[Induced Unary Operation]
Let $f:A\to A$. If
\[
a\sim a'\Longrightarrow f(a)\sim f(a'),
\]
then
\[
\overline{f}([a]):=[f(a)]
\]
defines a well-defined unary operation on $A/{\sim}$.
\end{theorem*}
