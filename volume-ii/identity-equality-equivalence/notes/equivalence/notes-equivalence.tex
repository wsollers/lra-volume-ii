\begin{definitionbox}{Definition (Equivalence Relation)}
\begin{definition}[Equivalence Relation]
\label{def:equivalence-relation}
Let $A$ be a set. A relation $\sim$ on $A$ is an \emph{equivalence relation}
if it is reflexive, symmetric, and transitive.
\end{definition}

\begin{remark*}[Standard quantified statement]
\[
\operatorname{EquivalenceRelation}_A(\sim)
\Longleftrightarrow
\Bigl(\forall x\in A,\ x\sim x\Bigr)
\land
\Bigl(\forall x,y\in A,\ x\sim y \Rightarrow y\sim x\Bigr)
\land
\Bigl(\forall x,y,z\in A,\ (x\sim y\land y\sim z)\Rightarrow x\sim z\Bigr).
\]
\end{remark*}

\begin{remark*}[Interpretation]
An equivalence relation formalizes sameness for a chosen purpose.
\end{remark*}

\NoLocalDependencies
\end{definitionbox}

\begin{corollarybox}{Corollary (Equality Is the Finest Equivalence)}
\begin{corollary}[Equality Is the Finest Equivalence]
\label{cor:equality-is-finest-equivalence}
Let $\sim$ be an equivalence relation on $A$. For all $a,b\in A$,
\[
a=b \Longrightarrow a\sim b.
\]
\end{corollary}
\end{corollarybox}

\begin{remark*}[Standard quantified statement]
\[
\forall a,b\in A,\,
a=b\Longrightarrow a\sim b.
\]
\end{remark*}

\begin{remark*}[Interpretation]
Equality is the finest equivalence relation because it identifies only objects
that are already the same. Any coarser equivalence relation must at least
identify equal representatives.
\end{remark*}

\begin{definitionbox}{Definition (Equivalence Class)}
\begin{definition}[Equivalence Class]
\label{def:equivalence-class}
Let $\sim$ be an equivalence relation on $A$. The \emph{equivalence class}
of $a\in A$ is
\[
[a] := \{x\in A \mid x\sim a\}.
\]
\end{definition}

\begin{remark*}[Standard quantified statement]
\[
x\in [a] \Longleftrightarrow x\sim a.
\]
\end{remark*}

\begin{remark*}[Interpretation]
The class $[a]$ collects all elements that are equivalent to the
representative $a$.
\end{remark*}

\begin{definitionbox}{Definition (Quotient Set)}
\begin{definition}[Quotient Set]
\label{def:quotient-set}
The \emph{quotient set} of $A$ by $\sim$ is the set of equivalence classes
\[
A/{\sim} := \{[a]\mid a\in A\}.
\]
\end{definition}

\begin{remark*}[Standard quantified statement]
\[
C\in A/{\sim} \Longleftrightarrow \exists a\in A,\ C=[a].
\]
\end{remark*}

\begin{remark*}[Interpretation]
The quotient replaces elements by their equivalence classes.
\end{remark*}

\begin{theorembox}{Theorem (Equivalence Classes Partition the Domain)}
\begin{theorem}[Equivalence Classes Partition the Domain]
\label{thm:equivalence-classes-partition-domain}
The equivalence classes of an equivalence relation on $A$ are nonempty, cover
$A$, and are pairwise disjoint.
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\forall a\in A,\ a\in[a],
\qquad
A=\bigcup_{a\in A}[a],
\qquad
[a]\cap[b]\ne\emptyset \Longrightarrow [a]=[b].
\]
\end{remark*}

\begin{remark*}[Interpretation]
Equivalence classes divide the domain into non-overlapping blocks.
\end{remark*}

\begin{propositionbox}{Proposition (Representatives Related iff Classes Equal)}
\begin{proposition}[Representatives Related iff Classes Equal]
\label{prop:representatives-related-iff-classes-equal}
\[
a\sim b \Longleftrightarrow [a]=[b].
\]
\end{proposition}
\end{propositionbox}

\begin{remark*}[Standard quantified statement]
\[
\forall a,b\in A,\ a\sim b \Longleftrightarrow [a]=[b].
\]
\end{remark*}

\begin{remark*}[Interpretation]
Changing representatives inside an equivalence class does not change the
class.
\end{remark*}

\subsection*{Canonical Projection}

\begin{definitionbox}{Definition (Canonical Projection)}
\begin{definition}[Canonical Projection]
\label{def:canonical-projection}
Let $\sim$ be an equivalence relation on $A$. The \emph{canonical projection}
is the map
\[
\pi:A\to A/{\sim},
\qquad
\pi(a):=[a].
\]
\end{definition}
\end{definitionbox}

\begin{propositionbox}{Proposition (Projection Is Surjective)}
\begin{proposition}[Projection Is Surjective]
\label{prop:projection-is-surjective}
The canonical projection $\pi:A\to A/{\sim}$ is surjective.
\end{proposition}
\end{propositionbox}

\begin{remark*}[Standard quantified statement]
\[
\forall C\in A/{\sim}\,\exists a\in A,\ \pi(a)=C.

\end{remark*}

\pi(a)=\pi(b)\Longleftrightarrow a\sim b.

\begin{theorem}[Existence of a Quotient]
\label{thm:existence-of-a-quotient}
For any equivalence relation $\sim$ on $A$, the quotient set $A/{\sim}$ exists
as the set of equivalence classes, and the canonical projection
\[
\pi:A\to A/{\sim}

\subsection*{Induced Structure on a Quotient}

\begin{definitionbox}{Definition (Operation Respects an Equivalence)}
\begin{definition}[Operation Respects an Equivalence]
\label{def:operation-respects-equivalence}
Let $O:A\times A\to A$ be a binary operation and let $\sim$ be an equivalence
relation on $A$. The operation $O$ \emph{respects} $\sim$ if
\[
\forall a,a',b,b'\in A,\,
\bigl(a\sim a'\land b\sim b'\bigr)
\Longrightarrow
\bigl(a\mathbin{O}b\bigr)\sim\bigl(a'\mathbin{O}b'\bigr).
\]
\end{definition}
\end{definitionbox}

\begin{definitionbox}{Definition (Left and Right Respect)}
\begin{definition}[Left and Right Respect]
\label{def:left-right-respect-equivalence}
The operation $O$ \emph{respects} $\sim$ on the left if
\[
a\sim a'\Longrightarrow (a\mathbin{O}b)\sim(a'\mathbin{O}b),
\]
and respects $\sim$ on the right if
\[
b\sim b'\Longrightarrow (a\mathbin{O}b)\sim(a\mathbin{O}b').
\]
\end{definition}
\end{definitionbox}

\begin{lemmabox}{Lemma (Respect Splits)}
\begin{lemma}[Respect Splits]
\label{lem:respect-splits}
A binary operation $O$ respects $\sim$ if and only if it respects $\sim$ on
the left and on the right.
\end{lemma}
\end{lemmabox}

\[
[a]\mathbin{\overline{O}}[b]:=[a\mathbin{O}b]

\begin{theorembox}{Theorem (Induced Map out of a Quotient)}
\begin{theorem}[Induced Map out of a Quotient]
\label{thm:induced-map-out-of-quotient}
Let $g:A\to B$. If
\[
a\sim a'\Longrightarrow g(a)=g(a'),
\]
then there is a unique well-defined map
\[
\overline{g}:A/{\sim}\to B
\]
such that
\[
\overline{g}([a])=g(a)
\qquad\text{and}\qquad
\overline{g}\circ\pi=g,
\]
where $\pi:A\to A/{\sim}$ is the canonical projection.
\end{theorem}
\end{theorembox}

\begin{theorembox}{Theorem (Induced Partial Operation on a Quotient)}
\begin{theorem}[Induced Partial Operation on a Quotient]
\label{thm:induced-partial-operation-on-quotient}
Let $D\subseteq A$ be saturated under $\sim$, meaning
\[
a\in D\land a\sim a'\Longrightarrow a'\in D.
\]
If $f:D\to A$ respects $\sim$ on $D$, then
\[
\overline{f}([a]):=[f(a)]
\]
is a well-defined partial operation on $A/{\sim}$ with domain
\[
D/{\sim}:=\{[a]\in A/{\sim}:a\in D\}.
\]
\end{theorem}
\end{theorembox}

\begin{lemmabox}{Lemma (Distinct Equivalence Classes Are Disjoint)}
\begin{lemma}[Distinct Equivalence Classes Are Disjoint]
\label{lem:distinct-equivalence-classes-are-disjoint}
\[
[a]\ne[b] \Longrightarrow [a]\cap[b]=\emptyset.
\]
\end{lemma}
\end{lemmabox}

\begin{remark*}[Standard quantified statement]
\[
\forall a,b\in A,\ [a]\ne[b] \Longrightarrow [a]\cap[b]=\emptyset.
\]
\end{remark*}

\begin{remark*}[Interpretation]
Two equivalence classes either coincide or do not overlap.
\end{remark*}
