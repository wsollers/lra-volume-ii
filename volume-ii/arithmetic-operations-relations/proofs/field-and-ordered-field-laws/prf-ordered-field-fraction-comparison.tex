\begin{theorem*}[Ordered Field Fraction Comparison]
In an ordered field, if $b\cdot d>0$, then
\[
\frac{a}{b}<\frac{c}{d}
\quad\Longleftrightarrow\quad
a\cdot d<c\cdot b.
\]
\end{theorem*}
