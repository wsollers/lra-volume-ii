\begin{theorem*}[Ordered Field Positive Multiplication Preserves Order]
In an ordered field,
\[
0<c\land a<b
\Longrightarrow
c\cdot a<c\cdot b.
\]
The corresponding non-strict implication also holds.
\end{theorem*}
