\begin{definitionbox}{Definition (Arithmetic Operation)}
\begin{definition}[Arithmetic Operation]
\label{def:arithmetic-operation}
Let $A$ be a domain. An \emph{arithmetic operation} on $A$ is a function
whose inputs and outputs lie in $A$. In particular, a binary arithmetic
operation is a function
\[
\star : A \times A \to A.
\]
\end{definition}

\begin{remark*}[Standard quantified statement]
\[
\operatorname{BinaryArithmeticOperation}_A(\star)
\Longleftrightarrow
\forall a,b \in A,\ \star(a,b) \in A.
\]
\end{remark*}

\begin{remark*}[Interpretation]
An arithmetic operation is a rule for producing new objects from old objects
without leaving the domain. Addition and multiplication are the central
examples: once they are defined on a number system, algebraic structure can be
studied through their closure, identity, associativity, and distributive
properties.
\end{remark*}

\end{definitionbox}

\begin{definitionbox}{Definition (Arithmetic Relation)}
\begin{definition}[Arithmetic Relation]
\label{def:arithmetic-relation}
Let $A$ be a domain. An \emph{arithmetic relation} on $A$ is a predicate on
tuples of elements of $A$. In particular, a binary arithmetic relation is a
subset
\[
R \subseteq A \times A.
\]
\end{definition}

\begin{remark*}[Standard quantified statement]
\[
\operatorname{BinaryArithmeticRelation}_A(R)
\Longleftrightarrow
\forall a,b,\ R(a,b) \text{ is a truth-valued assertion about } a,b \in A.
\]
\end{remark*}

\begin{remark*}[Interpretation]
An arithmetic relation does not create a new object. It compares objects and
returns a truth value. Equality, order, divisibility, and congruence are
relations; they organize the objects produced by arithmetic operations.
\end{remark*}

\end{definitionbox}

\begin{definitionbox}{Definition (Operation-Relation Compatibility)}
\begin{definition}[Operation-Relation Compatibility]
\label{def:operation-relation-compatibility}
Let $A$ be a domain with a binary operation $\star$ and a binary relation $R$.
The operation $\star$ is \emph{compatible} with $R$ if applying the operation
to related inputs preserves the relation in the intended coordinatewise sense.
\end{definition}

\begin{remark*}[Standard quantified statement]
\[
\operatorname{Compatible}_A(\star,R)
\Longleftrightarrow
\forall a,b,c,d \in A,\,
\bigl(R(a,b) \land R(c,d)\bigr)
\Rightarrow
R(a \star c, b \star d).
\]
\end{remark*}

\begin{remark*}[Interpretation]
Compatibility is the bridge between computation and comparison. It is the
reason addition can preserve order, multiplication can respect equality, and
operations can descend to quotient structures such as congruence classes or
fraction classes.
\end{remark*}

\end{definitionbox}
