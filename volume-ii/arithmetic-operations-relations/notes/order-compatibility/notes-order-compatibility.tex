\begin{definitionbox}{Definition (Right Monotony)}
\begin{definition}[Right Monotony]
\label{def:right-monotony}
Right monotony means
\[
yRz \Longrightarrow (x\mathbin{O}y)R(x\mathbin{O}z),
\]
\end{definition}
\end{definitionbox}

\begin{definitionbox}{Definition (Left Monotony)}
\begin{definition}[Left Monotony]
\label{def:left-monotony}
Left monotony means
\[
yRz \Longrightarrow (y\mathbin{O}x)R(z\mathbin{O}x).
\]
\end{definition}
\end{definitionbox}

\begin{definitionbox}{Definition (Two-Sided Monotony)}
\begin{definition}[Two-Sided Monotony]
\label{def:two-sided-monotony}
Two-sided monotony means both left monotony and right monotony hold.
\end{definition}
\end{definitionbox}

\begin{definitionbox}{Definition (Right Order Reflection)}
\begin{definition}[Right Order Reflection]
\label{def:right-order-reflection}
Right reflection means
\[
(x\mathbin{O}y)R(x\mathbin{O}z) \Longrightarrow yRz,
\]
\end{definition}
\end{definitionbox}

\begin{definitionbox}{Definition (Left Order Reflection)}
\begin{definition}[Left Order Reflection]
\label{def:left-order-reflection}
Left reflection means
\[
(y\mathbin{O}x)R(z\mathbin{O}x) \Longrightarrow yRz.
\]
\end{definition}
\end{definitionbox}

\begin{definitionbox}{Definition (Two-Sided Order Reflection)}
\begin{definition}[Two-Sided Order Reflection]
\label{def:two-sided-order-reflection}
Two-sided order reflection means both left order reflection and right order
reflection hold.
\end{definition}
\end{definitionbox}

\begin{definitionbox}{Definition (Monotone Map)}
\begin{definition}[Monotone Map]
\label{def:monotone-map}
\[
x\le y \Longrightarrow f(x)\le f(y).
\]
\end{definition}
\end{definitionbox}

\begin{definitionbox}{Definition (Strictly Monotone Map)}
\begin{definition}[Strictly Monotone Map]
\label{def:strictly-monotone-map}
\[
x<y \Longrightarrow f(x)<f(y).
\]
\end{definition}
\end{definitionbox}

\begin{propositionbox}{Proposition (Addition Preserves Order on the Left)}
\begin{proposition}[Addition Preserves Order on the Left]
\label{prop:addition-order-compatibility-left}
\[
y<z \Longrightarrow x+y<x+z
\]
in ordered arithmetic systems where addition is strictly order-preserving.
\end{proposition}
\end{propositionbox}

\begin{propositionbox}{Proposition (Addition Preserves Order)}
\begin{proposition}[Addition Preserves Order]
\label{prop:addition-order-compatibility}

addition preserves order on both sides.
\end{proposition}

\begin{propositionbox}{Proposition (Multiplication Preserves Order on the Left)}
\begin{proposition}[Multiplication Preserves Order on the Left]
\label{prop:multiplication-order-compatibility-left}
\[
0<x \land y<z \Longrightarrow x\cdot y<x\cdot z
\]
in ordered arithmetic systems where multiplication by positive elements is
strictly order-preserving.
\end{proposition}
