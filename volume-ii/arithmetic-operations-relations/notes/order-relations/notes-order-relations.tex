\begin{definitionbox}{Definition (Trichotomy)}
\begin{definition}[Trichotomy]
\label{def:trichotomy}
The relation $<$ satisfies \emph{trichotomy} if exactly one of
\[
x<y,\qquad x=y,\qquad y<x
\]
holds for every pair $x,y$.
\end{definition}
\end{definitionbox}
