\begin{definitionbox}{Definition (Preorder)}
\begin{definition}[Preorder]
\label{def:preorder}
A \emph{preorder} is a reflexive and transitive relation.
\end{definition}
\end{definitionbox}

\begin{definitionbox}{Definition (Partial Order)}
\begin{definition}[Partial Order]
\label{def:partial-order}
A \emph{partial order} is a preorder that is antisymmetric.
\end{definition}
\end{definitionbox}

\begin{definitionbox}{Definition (Strict Partial Order)}
\begin{definition}[Strict Partial Order]
\label{def:strict-partial-order}
A \emph{strict partial order} is an irreflexive and transitive relation.
\end{definition}
\end{definitionbox}

\begin{definitionbox}{Definition (Total Order)}
\begin{definition}[Total Order]
\label{def:total-order}
A \emph{total order} is a partial order whose relation is total.
\end{definition}
\end{definitionbox}

\begin{definitionbox}{Definition (Strict Total Order)}
\begin{definition}[Strict Total Order]
\label{def:strict-total-order}
A \emph{strict total order} is a strict partial order satisfying trichotomy.
\end{definition}
\end{definitionbox}

\begin{definitionbox}{Definition (Well-Order)}
\begin{definition}[Well-Order]
\label{def:well-order}
A \emph{well-order} is a total order in which every nonempty subset has a
least element.
\end{definition}
\end{definitionbox}
