\section{Axiom Systems for $\mathbb{N}$, $\mathbb{W}$, $\mathbb{Z}$, $\mathbb{Q}$, and $\mathbb{R}$}
\label{sec:number-systems-axioms}

\begin{toolkitbox}{Derived Properties — Quick Reference}
\small
\renewcommand{\arraystretch}{1.4}
\begin{tabular}{@{} l l p{0.42\linewidth} @{}}
\toprule
\textbf{Name} & \textbf{Statement} & \textbf{Proof sketch} \\
\midrule
Zero multiplication
  & $a \cdot 0 = 0$
  & $a\cdot 0 = a\cdot(0+0) = a\cdot 0 + a\cdot 0$;
    cancel $a\cdot 0$ via A4. \\
Negation rule
  & $(-a)\cdot b = -(ab)$
  & $ab + (-a)b = (a + (-a))b = 0\cdot b = 0$. \\
$(-1)$ rule
  & $(-1)\cdot a = -a$
  & Special case: $b = a$, $-a = (-1)a$. \\
Double negation
  & $(-a)(-b) = ab$
  & $(-a)(-b) = -(a(-b)) = -(-(ab)) = ab$. \\
No zero divisors
  & $ab = 0 \Rightarrow a=0$ or $b=0$
  & $\mathbb{Z}$ is an integral domain;
    proved via well-ordering + sign analysis. \\
\bottomrule
\end{tabular}
\end{toolkitbox}
