\begin{definitionbox}{Definition}
\begin{definition}[The Square-Root-of-Two Cut]
\label{def:sqrt-two-cut}
Define
\[
\sqrt{2}:=\{r\in\mathbb{Q}:r<0_{\mathbb{Q}}\lor r\cdot r<2_{\mathbb{Q}}\}.
\]
\end{definition}
\end{definitionbox}

\begin{lemmabox}{Lemma}
\begin{lemma}[The Square-Root-of-Two Cut Is a Cut]
\label{lem:sqrt-two-is-a-cut}
The set $\sqrt{2}$ is a Dedekind cut.
\end{lemma}

\begin{theorembox}{Theorem}
\begin{theorem}[Its Square Is Two]
\label{thm:sqrt-two-squared-is-two}
In $\mathbb{R}$,
\[
\sqrt{2}\cdot\sqrt{2}=2_{\mathbb{R}},
\]
where $2_{\mathbb{R}}:=\iota(2_{\mathbb{Q}})$.
\end{theorem}
\end{theorembox}

\begin{corollarybox}{Corollary}
\begin{corollary}[$\mathbb{R}$ Contains the Supremum $\mathbb{Q}$ Was Missing]
\label{cor:r-fills-the-gap}
Let
\[
S=\{q\in\mathbb{Q}:0_{\mathbb{Q}}<q\land q\cdot q<2_{\mathbb{Q}}\}.
\]
Then
\[
\sqrt{2}=\sup \iota(S).
\]
Thus the bounded-above rational set with no rational supremum has a supremum
in $\mathbb{R}$.
\end{corollary}
\end{corollarybox}
