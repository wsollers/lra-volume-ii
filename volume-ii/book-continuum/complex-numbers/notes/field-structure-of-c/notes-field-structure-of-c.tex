\subsection*{Additive Structure}

\begin{theorembox}{Theorem}
\begin{theorem}[Addition on $\mathbb{C}$ Is Associative]
\label{thm:addition-on-c-associative}
For all \(z,w,u\in\mathbb{C}\),
\[
(z+w)+u=z+(w+u).
\]
\end{theorem}
\end{theorembox}

\begin{theorembox}{Theorem}
\begin{theorem}[Addition on $\mathbb{C}$ Is Commutative]
\label{thm:addition-on-c-commutative}
For all \(z,w\in\mathbb{C}\),
\[
z+w=w+z.
\]
\end{theorem}
\end{theorembox}

\begin{theorembox}{Theorem}
\begin{theorem}[Zero Is an Additive Identity on $\mathbb{C}$]
\label{thm:zero-additive-identity-on-c}
For every \(z\in\mathbb{C}\),
\[
0_{\mathbb{C}}+z=z
\qquad\text{and}\qquad
z+0_{\mathbb{C}}=z.
\]
\end{theorem}
\end{theorembox}

\begin{theorembox}{Theorem}
\begin{theorem}[Every Complex Number Has an Additive Inverse]
\label{thm:every-complex-has-additive-inverse}
For every \(z\in\mathbb{C}\),
\[
z+(-z)=0_{\mathbb{C}}
\qquad\text{and}\qquad
(-z)+z=0_{\mathbb{C}}.
\]
\end{theorem}
\end{theorembox}

\subsection*{Multiplicative Structure}

\begin{theorembox}{Theorem}
\begin{theorem}[Multiplication on $\mathbb{C}$ Is Associative]
\label{thm:multiplication-on-c-associative}
For all \(z,w,u\in\mathbb{C}\),
\[
(zw)u=z(wu).
\]
\end{theorem}
\end{theorembox}

\begin{theorembox}{Theorem}
\begin{theorem}[Multiplication on $\mathbb{C}$ Is Commutative]
\label{thm:multiplication-on-c-commutative}
For all \(z,w\in\mathbb{C}\),
\[
zw=wz.
\]
\end{theorem}
\end{theorembox}

\begin{theorembox}{Theorem}
\begin{theorem}[One Is a Multiplicative Identity on $\mathbb{C}$]
\label{thm:one-multiplicative-identity-on-c}
For every \(z\in\mathbb{C}\),
\[
1_{\mathbb{C}}z=z
\qquad\text{and}\qquad
z1_{\mathbb{C}}=z.
\]
\end{theorem}
\end{theorembox}

\begin{theorembox}{Theorem}
\begin{theorem}[Every Nonzero Complex Number Has a Multiplicative Inverse]
\label{thm:every-nonzero-complex-has-inverse}
For every \(z\in\mathbb{C}\) with \(z\ne 0_{\mathbb{C}}\),
\[
zz^{-1}=1_{\mathbb{C}}
\qquad\text{and}\qquad
z^{-1}z=1_{\mathbb{C}}.
\]
\end{theorem}
\end{theorembox}

\subsection*{Distributivity and the Field Theorem}

\begin{theorembox}{Theorem}
\begin{theorem}[Multiplication Distributes over Addition on $\mathbb{C}$]
\label{thm:multiplication-distributes-over-addition-on-c}
For all \(z,w,u\in\mathbb{C}\),
\[
z(w+u)=zw+zu
\qquad\text{and}\qquad
(w+u)z=wz+uz.
\]
\end{theorem}
\end{theorembox}

\begin{theorembox}{Theorem}
\begin{theorem}[$\mathbb{C}$ Is a Field]
\label{thm:c-is-a-field}
With the operations defined above,
\[
(\mathbb{C},+,\,\cdot,\,0_{\mathbb{C}},\,1_{\mathbb{C}})
\]
is a field.
\end{theorem}
\end{theorembox}

\subsection*{The Imaginary Unit}

\begin{theorembox}{Theorem}
\begin{theorem}[The Imaginary Unit Squares to Negative One]
\label{thm:i-squared-negative-one}
In \(\mathbb{C}\),
\[
i^2=-1_{\mathbb{C}}.
\]
\end{theorem}
\end{theorembox}

\begin{remark*}[Interpretation]
The equation \(i^2=-1\) is the structural reason for introducing
\(\mathbb{C}\). It makes the polynomial \(x^2+1\) have a root, while the field
operations remain compatible with the real arithmetic already constructed.
\end{remark*}

\begin{theorembox}{Theorem}
\begin{theorem}[$\mathbb{C}$ Is Not an Ordered Field]
\label{thm:c-is-not-an-ordered-field}
There is no order relation on \(\mathbb{C}\) making \(\mathbb{C}\) an ordered
field and extending the usual order on \(\mathbb{R}\).
\end{theorem}
\end{theorembox}

\begin{remark*}[Reason]
In any ordered field, every square is nonnegative. If \(\mathbb{C}\) were an
ordered field extending the real order, \(i^2\) would be nonnegative. But
\(i^2=-1\), and \(-1<0\) in the real subfield.
\end{remark*}
