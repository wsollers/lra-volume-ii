\subsection*{Complex Prepairs}

\begin{topicbox}{The Last Number System Extension}
\begin{exposition}
The complex numbers extend the real numbers by adding a formal square root of
$-1$. Unlike the construction of $\mathbb{Z}$ from $\mathbb{W}$ or
$\mathbb{Q}$ from $\mathbb{Z}$, no nontrivial identification is needed:
ordered pairs of real numbers already have the correct equality. We still use
the same construction pattern: representatives, an equivalence relation,
equivalence classes, then operations whose well-definedness is checked before
they are used.
\end{exposition}
\end{topicbox}

\begin{definitionbox}{Definition}
\begin{definition}[Complex Prepair]
\label{def:complex-prepair}
A \emph{complex prepair} is an ordered pair $(a,b)\in\mathbb{R}\times\mathbb{R}$,
read informally as the expression $a+bi$.
\end{definition}
\end{definitionbox}

\subsection*{Coordinate Equivalence}

\begin{definitionbox}{Definition}
\begin{definition}[Complex Coordinate Equivalence]
\label{def:complex-coordinate-equivalence}
For $(a,b),(c,d)\in\operatorname{Pre}\mathbb{C}$, define
\[
(a,b)\sim_{\mathbb{C}}(c,d)
\quad:\Longleftrightarrow\quad
a=c\ \text{ and }\ b=d.
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Predicate reading]
\[
(a,b)\sim_{\mathbb{C}}(c,d)
\Longleftrightarrow
(a=c)\land(b=d).
\]
Thus the complex-number quotient is formally present, but the equivalence
classes are singleton coordinate-equality classes.
\end{remark*}

\begin{lemmabox}{Lemma}
\begin{lemma}[Complex Coordinate Equivalence Is Reflexive]
\label{lem:complex-coordinate-equivalence-reflexive}
For every $(a,b)\in\operatorname{Pre}\mathbb{C}$,
\[
(a,b)\sim_{\mathbb{C}}(a,b).
\]
\end{lemma}
\end{lemmabox}

\begin{lemmabox}{Lemma}
\begin{lemma}[Complex Coordinate Equivalence Is Symmetric]
\label{lem:complex-coordinate-equivalence-symmetric}
For all prepairs $(a,b),(c,d)$,
\[
(a,b)\sim_{\mathbb{C}}(c,d)
\Longrightarrow
(c,d)\sim_{\mathbb{C}}(a,b).
\]
\end{lemma}
\end{lemmabox}

\begin{lemmabox}{Lemma}
\begin{lemma}[Complex Coordinate Equivalence Is Transitive]
\label{lem:complex-coordinate-equivalence-transitive}
For all prepairs $(a,b),(c,d),(e,f)$,
\[
(a,b)\sim_{\mathbb{C}}(c,d)\land
(c,d)\sim_{\mathbb{C}}(e,f)
\Longrightarrow
(a,b)\sim_{\mathbb{C}}(e,f).
\]
\end{lemma}
\end{lemmabox}

\begin{theorembox}{Theorem}
\begin{theorem}[Complex Coordinate Equivalence Is an Equivalence Relation]
\label{thm:complex-coordinate-equivalence-relation}
The relation $\sim_{\mathbb{C}}$ is an equivalence relation on
\(\operatorname{Pre}\mathbb{C}\).
\end{theorem}
\end{theorembox}

\begin{remark*}[Proof sketch]
Reflexivity, symmetry, and transitivity are checked coordinate by coordinate
using equality in \(\mathbb{R}\). Thus this quotient construction has the same
formal shape as the constructions of \(\mathbb{Z}\) and \(\mathbb{Q}\), even
though the equivalence classes are singleton classes.
\end{remark*}

\subsection*{\texorpdfstring{Definition of $\mathbb{C}$}{Definition of C}}

\begin{definitionbox}{Definition}
\begin{definition}[Complex Numbers]
\label{def:complex-numbers}
The complex numbers are the quotient
\[
\mathbb{C}:=\operatorname{Pre}\mathbb{C}/{\sim_{\mathbb{C}}}.
\]
\end{definition}
\end{definitionbox}

\end{dependencies}

\begin{definitionbox}{Definition}
\begin{definition}[Complex Class Notation]
\label{def:complex-class-notation}
The notation $[a,b]_{\mathbb{C}}$ denotes the
\(\sim_{\mathbb{C}}\)-equivalence class of the prepair $(a,b)$.
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard form]
Since coordinate equivalence is equality of coordinates, every complex number
has a unique representative
\[
[a,b]_{\mathbb{C}}.
\]
We later write this same element as $a+bi$.
\end{remark*}

\begin{definitionbox}{Definition}
\begin{definition}[Zero, One, and the Imaginary Unit in $\mathbb{C}$]
\label{def:zero-one-i-in-c}
Define
\[
0_{\mathbb{C}}:=[0,0]_{\mathbb{C}},
\qquad
1_{\mathbb{C}}:=[1,0]_{\mathbb{C}},
\qquad
i:=[0,1]_{\mathbb{C}}.
\]
\end{definition}
\end{definitionbox}
