\begin{theorem*}[$Q$ Has Arbitrarily Close Distinct Points]
For every $x\in\mathbb{Q}$ and every $\varepsilon\in\mathbb{Q}$,
\[
0_{\mathbb{Q}}<\varepsilon
\Longrightarrow
\exists y\in\mathbb{Q}\ (y\ne x\ \text{and}\ |x-y|<\varepsilon).
\]
\end{theorem*}
