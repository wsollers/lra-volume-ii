\begin{theorem*}[Rational Square Approximation from Below]
Let $r\in\mathbb{Q}$ with $r>0$, and let $n\in\mathbb{N}$ with $n\ge 1$.
If $r-1/n>0$, then there exists $s\in\mathbb{Q}$ with $s>0$ such that
\[
r-\frac1n<s^2<r.
\]
\end{theorem*}
