\begin{theorem*}[Dyadic Approximation Error Bound]
For every $x\in\mathbb{Q}$ and every $n\in\mathbb{N}$, there exists
$m\in\mathbb{Z}$ such that
\[
\frac{m}{2^n}\le x<\frac{m+1}{2^n}.
\]
Consequently,
\[
0\le x-\frac{m}{2^n}<\frac{1}{2^n}.
\]
\end{theorem*}
