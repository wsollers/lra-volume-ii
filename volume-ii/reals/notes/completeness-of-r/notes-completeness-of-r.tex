\begin{definitionbox}{Definition}
\begin{definition}[Upper Bound in $\mathbb{R}$]
\label{def:upper-bound-in-r}
Let $S\subseteq\mathbb{R}$. A cut $\gamma\in\mathbb{R}$ is an upper bound of $S$
if
\[
\forall\alpha\in S\,(\alpha\le\gamma).
\]
The set $S$ is bounded above if it has an upper bound in $\mathbb{R}$.
\end{definition}
\end{definitionbox}

\begin{definitionbox}{Definition}
\begin{definition}[Supremum in $\mathbb{R}$]
\label{def:supremum-in-r}
For $S\subseteq\mathbb{R}$, a cut $\sigma$ is the supremum of $S$ if $\sigma$
is an upper bound of $S$ and every upper bound $\gamma$ of $S$ satisfies
$\sigma\le\gamma$.
\end{definition}
\end{definitionbox}

\begin{lemmabox}{Lemma}
\begin{lemma}[Union of a Bounded Nonempty Family of Cuts Is a Cut]
\label{lem:union-of-cuts-is-a-cut}
If $S\subseteq\mathbb{R}$ is nonempty and bounded above, then
\[
\bigcup_{\alpha\in S}\alpha
\]
is a Dedekind cut.
\end{lemma}
\end{lemmabox}

\begin{theorembox}{Theorem}
\begin{theorem}[Least-Upper-Bound Property]
\label{thm:lub-property-r}
Every nonempty subset $S\subseteq\mathbb{R}$ that is bounded above has a least
upper bound, namely
\[
\sup S=\bigcup_{\alpha\in S}\alpha.
\]
\end{theorem}
\end{theorembox}
