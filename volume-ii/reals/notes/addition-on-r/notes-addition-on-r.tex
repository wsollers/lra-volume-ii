\begin{definition}[Addition of Cuts]
\label{def:addition-on-r}
For cuts $\alpha,\beta\in\mathbb{R}$, define
\[
\alpha+\beta:=\{p+q:p\in\alpha,\ q\in\beta\}.
\]
\end{definition}

\begin{lemma}[Sum of Cuts Is a Cut]
\label{lem:sum-of-cuts-is-a-cut}
If $\alpha,\beta\in\mathbb{R}$, then $\alpha+\beta$ is a Dedekind cut.
\end{lemma}

\begin{theorem}[Addition on $\mathbb{R}$ Is Associative]
\label{thm:addition-on-r-associative}
For all $\alpha,\beta,\gamma\in\mathbb{R}$,
\[
(\alpha+\beta)+\gamma=\alpha+(\beta+\gamma).
\]
\end{theorem}

\begin{theorem}[Addition on $\mathbb{R}$ Is Commutative]
\label{thm:addition-on-r-commutative}
For all $\alpha,\beta\in\mathbb{R}$,
\[
\alpha+\beta=\beta+\alpha.
\]
\end{theorem}

\begin{theorem}[Zero Is an Additive Identity on $\mathbb{R}$]
\label{thm:zero-additive-identity-on-r}
For every $\alpha\in\mathbb{R}$,
\[
\alpha+0_{\mathbb{R}}=\alpha.
\]
\end{theorem}

\begin{definition}[Negation of a Cut]
\label{def:negation-on-r}
For $\alpha\in\mathbb{R}$, define
\[
-\alpha:=\{p\in\mathbb{Q}:\exists r\in\mathbb{Q}\,(r>0\land -p-r\notin\alpha)\}.
\]
\end{definition}

\begin{lemma}[Negation of a Cut Is a Cut]
\label{lem:negation-of-cut-is-a-cut}
If $\alpha\in\mathbb{R}$, then $-\alpha$ is a Dedekind cut.
\end{lemma}

\begin{theorem}[Negation Is an Additive Inverse on $\mathbb{R}$]
\label{thm:negation-additive-inverse-on-r}
For every $\alpha\in\mathbb{R}$,
\[
\alpha+(-\alpha)=0_{\mathbb{R}}.
\]
\end{theorem}

\begin{theorem}[$\mathbb{R}$ Is an Additive Abelian Group]
\label{thm:r-additive-abelian-group}
The structure $(\mathbb{R},+,0_{\mathbb{R}})$ is an abelian group.
\end{theorem}

\begin{definition}[Subtraction on $\mathbb{R}$]
\label{def:subtraction-on-r}
For cuts $\alpha,\beta\in\mathbb{R}$, define
\[
\alpha-\beta:=\alpha+(-\beta).
\]
\end{definition}
