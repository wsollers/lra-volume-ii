\subsection*{Order Compatibility}

\begin{theorembox}{Theorem}
\begin{theorem}[Addition Preserves Order on $\mathbb{R}$]
\label{thm:addition-preserves-order-on-r}
For all $\alpha,\beta,\gamma\in\mathbb{R}$,
\[
\alpha<\beta \Longrightarrow \alpha+\gamma<\beta+\gamma,
\]
and the corresponding non-strict statement also holds.
\end{theorem}
\end{theorembox}

\begin{theorembox}{Theorem}
\begin{theorem}[Multiplication by Positives Preserves Order on $\mathbb{R}$]
\label{thm:positive-multiplication-preserves-order-on-r}
For all $\alpha,\beta,\gamma\in\mathbb{R}$,
\[
0_{\mathbb{R}}<\gamma\land\alpha<\beta
\Longrightarrow
\gamma\cdot\alpha<\gamma\cdot\beta.
\]
\end{theorem}
\end{theorembox}

The structure $(\mathbb{R},+,\cdot,0_{\mathbb{R}},1_{\mathbb{R}},\le)$ is an
ordered field.
\end{theorem}
\end{theorembox}
