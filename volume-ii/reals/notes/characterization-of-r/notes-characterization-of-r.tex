\begin{definitionbox}{Definition}
\begin{definition}[Complete Ordered Field]
\label{def:real-ordered-field}
A \emph{complete ordered field} is an ordered field satisfying the
least-upper-bound property.
\end{definition}
\end{definitionbox}

\begin{theorembox}{Theorem}
\begin{theorem}[Uniqueness of the Complete Ordered Field]
\label{thm:uniqueness-of-cof}
Any two complete ordered fields are isomorphic by a unique order-isomorphism.
\end{theorem}
\end{theorembox}

\begin{corollarybox}{Corollary}
\begin{corollary}[$\mathbb{R}$ Is the Real Numbers]
\label{cor:r-is-the-reals}
The field $\mathbb{R}$ is determined up to unique order-isomorphism by being a
complete ordered field.
\end{corollary}
\end{corollarybox}
