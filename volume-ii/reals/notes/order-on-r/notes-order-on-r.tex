\subsection*{Order by Inclusion}

\begin{definitionbox}{Definition}
\begin{definition}[Order on $\mathbb{R}$]
\label{def:order-on-r}
For cuts $\alpha,\beta\in\mathbb{R}$, define
\[
\alpha\le\beta \quad:\Longleftrightarrow\quad \alpha\subseteq\beta.
\]
\end{definition}
\end{definitionbox}

\begin{dependencies}
\begin{itemize}
  \item \hyperref[def:reals]{The Real Numbers}
  \item \hyperref[def:subset]{Subset}
\end{itemize}
\end{dependencies}

\begin{definitionbox}{Definition}
\begin{definition}[Strict Order on $\mathbb{R}$]
\label{def:strict-order-on-r}
For cuts $\alpha,\beta\in\mathbb{R}$, define
\[
\alpha<\beta \quad:\Longleftrightarrow\quad \alpha\subsetneq\beta.
\]
\end{definition}
\end{definitionbox}

\begin{dependencies}
\begin{itemize}
  \item \hyperref[def:reals]{The Real Numbers}
  \item \hyperref[def:order-on-r]{Order on $\mathbb{R}$}
  \item \hyperref[def:proper-subset]{Proper Subset}
\end{itemize}
\end{dependencies}

\begin{theorembox}{Theorem}
\begin{theorem}[Order on $\mathbb{R}$ Is Reflexive]
\label{thm:order-on-r-reflexive}
For every $\alpha\in\mathbb{R}$, $\alpha\le\alpha$.
\end{theorem}
\end{theorembox}

\begin{theorembox}{Theorem}
\begin{theorem}[Order on $\mathbb{R}$ Is Antisymmetric]
\label{thm:order-on-r-antisymmetric}
For all $\alpha,\beta\in\mathbb{R}$,
\[
\alpha\le\beta\land\beta\le\alpha \Longrightarrow \alpha=\beta.
\]
\end{theorem}
\end{theorembox}

\begin{theorembox}{Theorem}
\begin{theorem}[Order on $\mathbb{R}$ Is Transitive]
\label{thm:order-on-r-transitive}
For all $\alpha,\beta,\gamma\in\mathbb{R}$,
\[
\alpha\le\beta\land\beta\le\gamma \Longrightarrow \alpha\le\gamma.
\]
\end{theorem}
\end{theorembox}

\begin{theorembox}{Theorem}
\begin{theorem}[Comparability of Cuts]
\label{thm:comparability-of-cuts}
For all cuts $\alpha,\beta\in\mathbb{R}$,
\[
\alpha\subseteq\beta \lor \beta\subseteq\alpha.
\]
\end{theorem}
\end{theorembox}

\begin{theorembox}{Theorem}
\begin{theorem}[Order on $\mathbb{R}$ Is Total]
\label{thm:order-on-r-total}
For all $\alpha,\beta\in\mathbb{R}$,
\[
\alpha\le\beta\lor\beta\le\alpha.
\]
\end{theorem}
\end{theorembox}

\begin{theorembox}{Theorem}
\begin{theorem}[Trichotomy on $\mathbb{R}$]
\label{thm:trichotomy-on-r}
For all $\alpha,\beta\in\mathbb{R}$, exactly one of
\[
\alpha<\beta,\qquad \alpha=\beta,\qquad \beta<\alpha
\]
holds.
\end{theorem}
\end{theorembox}

\begin{theorembox}{Theorem}
\begin{theorem}[$\mathbb{R}$ Is a Totally Ordered Set]
\label{thm:r-totally-ordered-set}
The ordered pair $(\mathbb{R},\le)$ is a total order.
\end{theorem}
\end{theorembox}
