\section{Intervals in the Real Numbers}

% ---------------------------------------------------------
% Toolkit
% ---------------------------------------------------------
\begin{exposition}
Intervals are fundamental subsets of the real line defined using the order
relation on $\mathbb{R}$. Let $a,b \in \mathbb{R}$ with $a < b$.
\end{exposition}

% ---------------------------------------------------------
\subsection*{Basic Definitions}
% ---------------------------------------------------------

\section*{Bounded Intervals}

\begin{definitionbox}{Definition (Open Interval)}
The \emph{open interval} from $a$ to $b$ is the set
\[
(a,b) := \{ x \in \mathbb{R} : a < x < b \}.
\]
\[
\forall x \, (x \in (a,b) \leftrightarrow a < x \wedge x < b)
\]
\end{definitionbox}
\label{def:open-interval-R}

\begin{definitionbox}{Definition (Closed Interval)}
The \emph{closed interval} from $a$ to $b$ is the set
\[
[a,b] := \{ x \in \mathbb{R} : a \le x \le b \}.
\]
\[
\forall x \, (x \in [a,b] \leftrightarrow a \le x \wedge x \le b)
\]
\end{definitionbox}
\label{def:closed-interval-R}

\begin{definitionbox}{Definition (Half-Open Intervals)}
The \emph{left-closed, right-open interval} is
\[
[a,b) := \{ x \in \mathbb{R} : a \le x < b \}.
\]
The \emph{left-open, right-closed interval} is
\[
(a,b] := \{ x \in \mathbb{R} : a < x \le b \}.
\]
\end{definitionbox}

\section*{Unbounded Intervals}

\begin{definitionbox}{Definition (Open Rays)}
The \emph{open rays} determined by $a \in \mathbb{R}$ are
\[
(a,\infty) := \{ x \in \mathbb{R} : x > a \},
\qquad
(-\infty,a) := \{ x \in \mathbb{R} : x < a \}.
\]
\end{definitionbox}

\begin{definitionbox}{Definition (Closed Rays)}
The \emph{closed rays} determined by $a \in \mathbb{R}$ are
\[
[a,\infty) := \{ x \in \mathbb{R} : x \ge a \},
\qquad
(-\infty,a] := \{ x \in \mathbb{R} : x \le a \}.
\]
\end{definitionbox}

\begin{theorembox}{Theorem (Upper/Lower Perturbation Lemma in $\mathbb{R}$)}
\begin{theorem}[Upper/Lower Perturbation Lemma in $\mathbb{R}$]
\label{thm:real-upper-lower-perturbation}
If $x\in\mathbb{R}$ and $\varepsilon\in\mathbb{R}$ with $\varepsilon>0$, then
there exist $r,s\in\mathbb{R}$ such that
\[
x-\varepsilon<r<x<s<x+\varepsilon.
\]
\end{theorem}
\end{theorembox}

as part of its ordered-field structure.
\end{dependencies}

Take the symmetric perturbation $\delta=\varepsilon/2$ and set
$r=x-\delta$, $s=x+\delta$.
\end{remark*}

\section*{Degenerate and Trivial Intervals}

\begin{definitionbox}{Definition (Degenerate Interval)}
If $a=b$, the closed interval
\[
[a,a] = \{a\}
\]
is called a \emph{degenerate interval}.
\end{definitionbox}

\begin{definitionbox}{Definition (Empty Interval)}
If $a>b$, the set
\[
(a,b) = \varnothing
\]
is called an \emph{empty interval}.
\end{definitionbox}

% ---------------------------------------------------------
\subsection*{Main Theorems}
% ---------------------------------------------------------

\begin{theorem}[Convexity Characterization of Intervals in $\mathbb{R}$]
\label{thm:convexity-characterization-of-intervals-in-r}
A subset $I \subseteq \mathbb{R}$ is an interval if and only if
whenever $x,z \in I$ and $x < y < z$, then $y \in I$.
Equivalently,
\[
\forall x,z \in I,\; x<z \Rightarrow
\bigl(
\forall y \in \mathbb{R},\;
(x<y<z \Rightarrow y \in I)
\bigr).
\]
\hyperref[prf:convexity-characterization-of-intervals-in-r]{Go to proof.}
\end{theorem}



\begin{remark*}
Intervals are precisely the subsets of $\mathbb{R}$ with the property that if
$x<y<z$ and $x,z$ belong to the set, then $y$ also belongs to the set.
\end{remark*}

\subsection*{Consequences}
% ---------------------------------------------------------

\begin{exposition}
The logical implication of this section is:
\end{exposition}

\begin{exposition}
\[
\text{Order Structure of }\mathbb{R}
\Rightarrow
\text{Intervals}
\Rightarrow
\text{Convex subsets of }\mathbb{R}.
\]
\end{exposition}

\begin{exposition}
Intervals encode the idea of ``no gaps'' between points of a set.
They are the basic building blocks for:
\end{exposition}

\begin{itemize}
\item neighborhoods,
\item open sets,
\item topology of $\mathbb{R}$,
\item and completeness arguments (e.g. nested intervals).
\end{itemize}

\end{remark*}
