\begin{definition}[Nonnegative Cut]
\label{def:nonnegative-cut}
A cut $\alpha\in\mathbb{R}$ is nonnegative if $0_{\mathbb{R}}\le\alpha$ and
positive if $0_{\mathbb{R}}<\alpha$.
\end{definition}

\begin{definition}[Product of Nonnegative Cuts]
\label{def:product-nonnegative-on-r}
For nonnegative cuts $\alpha,\beta\in\mathbb{R}$, define
\[
\alpha\cdot\beta
:=
\{r\in\mathbb{Q}:r<0_{\mathbb{Q}}\}
\cup
\{p\cdot q:p\in\alpha,\ q\in\beta,\ p\ge 0_{\mathbb{Q}},\ q\ge 0_{\mathbb{Q}}\}.
\]
\end{definition}

\begin{lemma}[Product of Nonnegative Cuts Is a Cut]
\label{lem:product-nonnegative-is-a-cut}
If $\alpha,\beta\ge 0_{\mathbb{R}}$, then $\alpha\cdot\beta$ is a Dedekind cut.
\end{lemma}

\begin{definition}[Absolute Value on $\mathbb{R}$]
\label{def:real-chapter-absolute-value-on-r}
For $\alpha\in\mathbb{R}$, define
\[
|\alpha|:=
\begin{cases}
\alpha, & 0_{\mathbb{R}}\le\alpha,\\
-\alpha, & \alpha<0_{\mathbb{R}}.
\end{cases}
\]
\end{definition}

\begin{definition}[Multiplication of Cuts]
\label{def:multiplication-on-r}
For cuts $\alpha,\beta\in\mathbb{R}$, define $\alpha\cdot\beta$ by sign cases:
use $|\alpha|\cdot|\beta|$ when $\alpha$ and $\beta$ have the same sign, use
$-(|\alpha|\cdot|\beta|)$ when they have opposite signs, and use
$0_{\mathbb{R}}$ when either factor is $0_{\mathbb{R}}$.
\end{definition}

\begin{lemma}[Product of Cuts Is a Cut]
\label{lem:product-of-cuts-is-a-cut}
If $\alpha,\beta\in\mathbb{R}$, then $\alpha\cdot\beta\in\mathbb{R}$.
\end{lemma}

\begin{theorem}[Multiplication on $\mathbb{R}$ Is Associative]
\label{thm:multiplication-on-r-associative}
For all $\alpha,\beta,\gamma\in\mathbb{R}$,
\[
(\alpha\cdot\beta)\cdot\gamma=\alpha\cdot(\beta\cdot\gamma).
\]
\end{theorem}

\begin{theorem}[Multiplication on $\mathbb{R}$ Is Commutative]
\label{thm:multiplication-on-r-commutative}
For all $\alpha,\beta\in\mathbb{R}$,
\[
\alpha\cdot\beta=\beta\cdot\alpha.
\]
\end{theorem}

\begin{theorem}[One Is a Multiplicative Identity on $\mathbb{R}$]
\label{thm:one-multiplicative-identity-on-r}
For every $\alpha\in\mathbb{R}$,
\[
\alpha\cdot 1_{\mathbb{R}}=\alpha.
\]
\end{theorem}

\begin{definition}[Reciprocal of a Positive Cut]
\label{def:reciprocal-positive-on-r}
For $\alpha>0_{\mathbb{R}}$, define
\[
\alpha^{-1}:=\{r\in\mathbb{Q}:r\le 0_{\mathbb{Q}}\}
\cup
\{r\in\mathbb{Q}:r>0_{\mathbb{Q}}\land \exists s\in\mathbb{Q}\,(s>r\land s^{-1}\notin\alpha)\}.
\]
\end{definition}

\begin{definition}[Reciprocal of a Nonzero Cut]
\label{def:reciprocal-on-r}
For $\alpha\ne0_{\mathbb{R}}$, define $\alpha^{-1}$ to be $|\alpha|^{-1}$ if
$\alpha>0_{\mathbb{R}}$ and $-(|\alpha|^{-1})$ if $\alpha<0_{\mathbb{R}}$.
\end{definition}

\begin{lemma}[Reciprocal of a Nonzero Cut Is a Cut]
\label{lem:reciprocal-is-a-cut}
If $\alpha\ne0_{\mathbb{R}}$, then $\alpha^{-1}\in\mathbb{R}$.
\end{lemma}

\begin{theorem}[Reciprocal Is a Multiplicative Inverse on $\mathbb{R}$]
\label{thm:reciprocal-multiplicative-inverse-on-r}
If $\alpha\ne0_{\mathbb{R}}$, then
\[
\alpha\cdot\alpha^{-1}=1_{\mathbb{R}}.
\]
\end{theorem}

\begin{definition}[Division on $\mathbb{R}$]
\label{def:division-on-r}
For $\beta\ne0_{\mathbb{R}}$, define
\[
\alpha/\beta:=\alpha\cdot\beta^{-1}.
\]
\end{definition}

\begin{theorem}[Distributivity on $\mathbb{R}$]
\label{thm:distributivity-on-r}
For all $\alpha,\beta,\gamma\in\mathbb{R}$,
\[
\alpha\cdot(\beta+\gamma)=\alpha\cdot\beta+\alpha\cdot\gamma.
\]
\end{theorem}
