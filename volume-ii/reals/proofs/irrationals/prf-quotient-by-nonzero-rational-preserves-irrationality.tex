\begin{theorem*}[Quotient by Nonzero Rational Preserves Irrationality]
If $\alpha\in\mathbb{R}\setminus\mathbb{Q}$ and
$q\in\mathbb{Q}\setminus\{0\}$, then
\[
\frac{\alpha}{q}\in\mathbb{R}\setminus\mathbb{Q}.
\]
\end{theorem*}
