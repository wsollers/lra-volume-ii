\subsection*{\texorpdfstring{Addition on $\mathbb{C}$-Classes}{Addition on C-Classes}}

\begin{definitionbox}{Definition}
\begin{definition}[Addition on $\mathbb{C}$-Classes]
\label{def:addition-on-c-classes}
Define addition on representatives by
\[
[a,b]_{\mathbb{C}}+[c,d]_{\mathbb{C}}
:=[a+c,b+d]_{\mathbb{C}}.
\]
\end{definition}
\end{definitionbox}

\begin{dependencies}
\begin{itemize}
  \item \hyperref[def:complex-class-notation]{Complex Class Notation}
  \item \hyperref[def:addition-on-r]{Addition on $\mathbb{R}$}
\end{itemize}
\end{dependencies}

\begin{lemmabox}{Lemma}
\begin{lemma}[Addition Respects Complex Coordinate Equivalence]
\label{lem:addition-on-c-respects-equivalence}
If $[a,b]_{\mathbb{C}}=[a',b']_{\mathbb{C}}$ and
$[c,d]_{\mathbb{C}}=[c',d']_{\mathbb{C}}$, then
\[
[a,b]_{\mathbb{C}}+[c,d]_{\mathbb{C}}
=
[a',b']_{\mathbb{C}}+[c',d']_{\mathbb{C}}.
\]
\end{lemma}
\end{lemmabox}

\begin{dependencies}
\begin{itemize}
  \item \hyperref[def:addition-on-c-classes]{Addition on $\mathbb{C}$}
  \item \hyperref[def:complex-coordinate-equivalence]{Complex Coordinate Equivalence}
\end{itemize}
\end{dependencies}

\begin{theorembox}{Theorem}
\begin{theorem}[Addition on $\mathbb{C}$ Is Well-Defined]
\label{thm:addition-on-c-well-defined}
Addition determines a well-defined binary operation
\[
\mathbb{C}\times\mathbb{C}\to\mathbb{C}.
\]
\end{theorem}
\end{theorembox}

\begin{remark*}[Proof sketch]
Suppose the input classes are represented in two ways. Coordinate equivalence
gives \(a=a'\), \(b=b'\), \(c=c'\), and \(d=d'\). Addition in
\(\mathbb{R}\) gives \(a+c=a'+c'\) and \(b+d=b'+d'\), so the resulting
complex classes agree.
\end{remark*}

\subsection*{Negation and Subtraction}

\begin{definitionbox}{Definition}
\begin{definition}[Negation on $\mathbb{C}$-Classes]
\label{def:negation-on-c-classes}
Define negation by
\[
-[a,b]_{\mathbb{C}}:=[-a,-b]_{\mathbb{C}}.
\]
\end{definition}
\end{definitionbox}

\begin{lemmabox}{Lemma}
\begin{lemma}[Negation Respects Complex Coordinate Equivalence]
\label{lem:negation-on-c-respects-equivalence}
If $[a,b]_{\mathbb{C}}=[a',b']_{\mathbb{C}}$, then
\[
-[a,b]_{\mathbb{C}}=-[a',b']_{\mathbb{C}}.
\]
\end{lemma}
\end{lemmabox}

\begin{theorembox}{Theorem}
\begin{theorem}[Negation on $\mathbb{C}$ Is Well-Defined]
\label{thm:negation-on-c-well-defined}
Negation determines a well-defined unary operation $\mathbb{C}\to\mathbb{C}$.
\end{theorem}
\end{theorembox}

\begin{dependencies}
\begin{itemize}
  \item \hyperref[def:function]{Function}
  \item \hyperref[def:negation-on-c-classes]{Negation on $\mathbb{C}$}
  \item \hyperref[lem:negation-on-c-respects-equivalence]{Negation Respects Complex Coordinate Equivalence}
\end{itemize}
\end{dependencies}

\begin{remark*}[Proof sketch]
If \([a,b]_{\mathbb{C}}=[a',b']_{\mathbb{C}}\), then \(a=a'\) and
\(b=b'\). Hence \(-a=-a'\) and \(-b=-b'\), so the negated representatives
determine the same complex class.
\end{remark*}

\begin{definitionbox}{Definition}
\begin{definition}[Subtraction on $\mathbb{C}$]
\label{def:subtraction-on-c}
For $z,w\in\mathbb{C}$, define
\[
z-w:=z+(-w).
\]
\end{definition}
\end{definitionbox}

\begin{theorembox}{Theorem}
\begin{theorem}[Subtraction on $\mathbb{C}$ Is Well-Defined]
\label{thm:subtraction-on-c-well-defined}
Subtraction determines a well-defined binary operation
\(\mathbb{C}\times\mathbb{C}\to\mathbb{C}\).
\end{theorem}
\end{theorembox}

\subsection*{\texorpdfstring{Multiplication on $\mathbb{C}$-Classes}{Multiplication on C-Classes}}

\begin{definitionbox}{Definition}
\begin{definition}[Multiplication on $\mathbb{C}$-Classes]
\label{def:multiplication-on-c-classes}
Define multiplication on representatives by
\[
[a,b]_{\mathbb{C}}\cdot[c,d]_{\mathbb{C}}
:=[ac-bd,ad+bc]_{\mathbb{C}}.
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Why this formula]
The rule is forced by the intended identity \(i^2=-1\):
\[
(a+bi)(c+di)=(ac-bd)+(ad+bc)i.
\]
\end{remark*}

\begin{dependencies}
\begin{itemize}
  \item \hyperref[def:complex-class-notation]{Complex Class Notation}
  \item \hyperref[def:multiplication-on-r]{Multiplication on $\mathbb{R}$}
  \item \hyperref[def:addition-on-r]{Addition on $\mathbb{R}$}
\end{itemize}
\end{dependencies}

\begin{lemmabox}{Lemma}
\begin{lemma}[Multiplication Respects Complex Coordinate Equivalence]
\label{lem:multiplication-on-c-respects-equivalence}
If $[a,b]_{\mathbb{C}}=[a',b']_{\mathbb{C}}$ and
$[c,d]_{\mathbb{C}}=[c',d']_{\mathbb{C}}$, then
\[
[a,b]_{\mathbb{C}}\cdot[c,d]_{\mathbb{C}}
=
[a',b']_{\mathbb{C}}\cdot[c',d']_{\mathbb{C}}.
\]
\end{lemma}
\end{lemmabox}

\begin{theorembox}{Theorem}
\begin{theorem}[Multiplication on $\mathbb{C}$ Is Well-Defined]
\label{thm:multiplication-on-c-well-defined}
Multiplication determines a well-defined binary operation
\[
\mathbb{C}\times\mathbb{C}\to\mathbb{C}.
\]
\end{theorem}
\end{theorembox}

\begin{remark*}[Proof sketch]
Changing representatives preserves each coordinate. Therefore the real
expressions \(ac-bd\) and \(ad+bc\) are unchanged after replacing
\((a,b)\) by an equivalent prepair and \((c,d)\) by an equivalent prepair.
The product class is independent of the chosen representatives.
\end{remark*}

\subsection*{Conjugation and Inverses}

\begin{definitionbox}{Definition}
\begin{definition}[Complex Conjugation]
\label{def:complex-conjugation}
For \(z=[a,b]_{\mathbb{C}}\), define its \emph{complex conjugate} by
\[
\overline{z}:=[a,-b]_{\mathbb{C}}.
\]
\end{definition}
\end{definitionbox}

\begin{theorembox}{Theorem}
\begin{theorem}[Complex Conjugation Is Well-Defined]
\label{thm:complex-conjugation-well-defined}
Complex conjugation determines a well-defined unary operation
\(\mathbb{C}\to\mathbb{C}\).
\end{theorem}
\end{theorembox}

\begin{dependencies}
\begin{itemize}
  \item \hyperref[def:function]{Function}
  \item \hyperref[def:complex-conjugation]{Complex Conjugation}
\end{itemize}
\end{dependencies}

\begin{remark*}[Proof sketch]
Equivalent prepairs have equal first coordinates and equal second coordinates.
Keeping the first coordinate and negating the second preserves that equality,
so conjugation does not depend on the representative.
\end{remark*}

\begin{lemmabox}{Lemma}
\begin{lemma}[Product with the Conjugate Is Real and Nonnegative]
\label{lem:complex-product-with-conjugate}
For \(z=[a,b]_{\mathbb{C}}\),
\[
z\overline{z}=[a^2+b^2,0]_{\mathbb{C}}.
\]
Moreover, if \(z\ne 0_{\mathbb{C}}\), then \(a^2+b^2\ne 0\).
\end{lemma}
\end{lemmabox}

\begin{definitionbox}{Definition}
\begin{definition}[Multiplicative Inverse in $\mathbb{C}$]
\label{def:multiplicative-inverse-on-c}
For \(z=[a,b]_{\mathbb{C}}\ne 0_{\mathbb{C}}\), define
\[
z^{-1}
:=
\left[\frac{a}{a^2+b^2},\frac{-b}{a^2+b^2}\right]_{\mathbb{C}}.
\]
\end{definition}
\end{definitionbox}

\begin{theorembox}{Theorem}
\begin{theorem}[Inversion on $\mathbb{C}^{\times}$ Is Well-Defined]
\label{thm:inversion-on-c-well-defined}
Inversion determines a well-defined operation on the nonzero complex numbers.
\end{theorem}
\end{theorembox}

\begin{dependencies}
\begin{itemize}
  \item \hyperref[def:function]{Function}
  \item \hyperref[def:multiplicative-inverse-on-c]{Multiplicative Inverse in $\mathbb{C}$}
  \item \hyperref[lem:complex-product-with-conjugate]{Product with the Conjugate Is Real and Nonnegative}
\end{itemize}
\end{dependencies}

\begin{remark*}[Proof sketch]
If \(z=[a,b]_{\mathbb{C}}\ne0_{\mathbb{C}}\), then \(a\) and \(b\) are not
both zero, so \(a^2+b^2\ne0\) in \(\mathbb{R}\). Equivalent representatives
have the same coordinates, hence the same denominator and the same inverse
coordinates.
\end{remark*}
