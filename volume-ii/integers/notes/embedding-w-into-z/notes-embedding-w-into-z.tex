\subsection*{Embedding $\mathbb{W}$ into $\mathbb{Z}$}

\begin{definition}[Embedding of $\mathbb{W}$ into $\mathbb{Z}$]
\label{def:canonical-embedding-w-into-z}
Define
\[
\iota:\mathbb{W}\to\mathbb{Z},
\qquad
\iota(n):=[n,0].
\]
\end{definition}

\begin{theorem}[Embedding Preserves Zero]
\label{thm:embedding-w-into-z-preserves-zero}
The canonical embedding satisfies
\[
\iota(0_{\mathbb{W}})=0_{\mathbb{Z}}.
\]
\end{theorem}

\begin{theorem}[Embedding Preserves One]
\label{thm:embedding-w-into-z-preserves-one}
The canonical embedding satisfies
\[
\iota(1_{\mathbb{W}})=1_{\mathbb{Z}}.
\]
\end{theorem}

\begin{theorem}[Embedding $\mathbb{W}$ into $\mathbb{Z}$ Is Injective]
\label{thm:embedding-w-into-z-injective}
For all $a,b\in\mathbb{W}$,
\[
\iota(a)=\iota(b)\Longrightarrow a=b.
\]
\end{theorem}

\begin{theorem}[Embedding Preserves Addition]
\label{thm:embedding-w-into-z-preserves-addition}
For all $a,b\in\mathbb{W}$,
\[
\iota(a+b)=\iota(a)+\iota(b).
\]
\end{theorem}

\begin{theorem}[Embedding Preserves Multiplication]
\label{thm:embedding-w-into-z-preserves-multiplication}
For all $a,b\in\mathbb{W}$,
\[
\iota(a\cdot b)=\iota(a)\cdot\iota(b).
\]
\end{theorem}

\begin{theorem}[Embedding Preserves Order]
\label{thm:embedding-w-into-z-preserves-order}
For all $a,b\in\mathbb{W}$,
\[
a\le b \Longleftrightarrow \iota(a)\le\iota(b).
\]
\end{theorem}

\begin{theorem}[$\mathbb{W}$ Embeds into $\mathbb{Z}$]
\label{thm:w-embeds-into-z}
The map $\iota:\mathbb{W}\to\mathbb{Z}$ is an injective order-preserving
semiring homomorphism.
\end{theorem}
