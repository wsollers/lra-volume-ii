\subsection*{\texorpdfstring{Embedding $\mathbb{W}$ into $\mathbb{Z}$}{Embedding W into Z}}

\begin{definitionbox}{Definition}
\begin{definition}[Embedding of $\mathbb{W}$ into $\mathbb{Z}$]
\label{def:canonical-embedding-w-into-z}
Define
\[
\iota:\mathbb{W}\to\mathbb{Z},
\qquad
\iota(n):=[n,0].
\]
\end{definition}
\end{definitionbox}

\iota(a)=\iota(b)\Longrightarrow a=b.
