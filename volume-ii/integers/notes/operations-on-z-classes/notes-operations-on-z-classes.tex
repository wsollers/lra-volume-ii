\subsection*{Addition on $\mathbb{Z}$-Classes}

\begin{definitionbox}{Definition}
\begin{definition}[Addition on $\mathbb{Z}$-Classes]
\label{def:addition-on-z-classes}
Define addition on representatives by
\[
[a,b]+[c,d]:=[a+c,b+d].
\]
\end{definition}
\end{definitionbox}

\begin{lemmabox}{Lemma}
\begin{lemma}[Addition Respects Formal-Difference Equivalence on the Left]
\label{lem:addition-on-z-classes-respects-left-equivalence}
If $[a,b]=[a',b']$, then
\[
[a,b]+[c,d]=[a',b']+[c,d].
\]
\end{lemma}
\end{lemmabox}

\begin{lemmabox}{Lemma}
\begin{lemma}[Addition Respects Formal-Difference Equivalence on the Right]
\label{lem:addition-on-z-classes-respects-right-equivalence}
If $[c,d]=[c',d']$, then
\[
[a,b]+[c,d]=[a,b]+[c',d'].
\]
\end{lemma}
\end{lemmabox}

\begin{lemmabox}{Lemma}
\begin{lemma}[Addition Respects Formal-Difference Equivalence]
\label{lem:addition-on-z-classes-respects-equivalence}
If $[a,b]=[a',b']$ and $[c,d]=[c',d']$, then
\[
[a,b]+[c,d]=[a',b']+[c',d'].
\]
\end{lemma}
\end{lemmabox}

\begin{theorembox}{Theorem}
\begin{theorem}[Addition on $\mathbb{Z}$ Is Well-Defined]
\label{thm:addition-on-z-well-defined}
Addition determines a well-defined binary operation
$\mathbb{Z}\times\mathbb{Z}\to\mathbb{Z}$.
\end{theorem}
\end{theorembox}

\begin{theorembox}{Theorem}
\begin{theorem}[Addition on $\mathbb{Z}$ Is Associative]
\label{thm:addition-on-z-associative}
For all $x,y,z\in\mathbb{Z}$,
\[
(x+y)+z=x+(y+z).
\]
\end{theorem}
\end{theorembox}

\begin{theorembox}{Theorem}
\begin{theorem}[Addition on $\mathbb{Z}$ Is Commutative]
\label{thm:addition-on-z-commutative}
For all $x,y\in\mathbb{Z}$,
\[
x+y=y+x.
\]
\end{theorem}
\end{theorembox}

\begin{theorembox}{Theorem}
\begin{theorem}[Zero Is a Left Additive Identity on $\mathbb{Z}$]
\label{thm:zero-left-additive-identity-on-z}
For every $x\in\mathbb{Z}$,
\[
0_{\mathbb{Z}}+x=x.
\]
\end{theorem}
\end{theorembox}

\begin{corollarybox}{Corollary}
\begin{corollary}[Zero Is a Right Additive Identity on $\mathbb{Z}$]
\label{cor:zero-right-additive-identity-on-z}
For every $x\in\mathbb{Z}$,
\[
x+0_{\mathbb{Z}}=x.
\]
\end{corollary}
\end{corollarybox}

\begin{theorembox}{Theorem}
\begin{theorem}[Zero Is an Additive Identity on $\mathbb{Z}$]
\label{thm:zero-additive-identity-on-z}
For every $x\in\mathbb{Z}$,
\[
0_{\mathbb{Z}}+x=x
\qquad\text{and}\qquad
x+0_{\mathbb{Z}}=x.
\]
\end{theorem}
\end{theorembox}

\subsection*{Negation on $\mathbb{Z}$-Classes}

\begin{definitionbox}{Definition}
\begin{definition}[Negation on $\mathbb{Z}$-Classes]
\label{def:negation-on-z-classes}
Define negation on representatives by
\[
-[a,b]:=[b,a].
\]
\end{definition}
\end{definitionbox}

\begin{lemmabox}{Lemma}
\begin{lemma}[Negation Respects Formal-Difference Equivalence]
\label{lem:negation-on-z-classes-respects-equivalence}
If $[a,b]=[a',b']$, then
\[
-[a,b]=-[a',b'].
\]
\end{lemma}
\end{lemmabox}

\begin{theorembox}{Theorem}
\begin{theorem}[Negation on $\mathbb{Z}$ Is Well-Defined]
\label{thm:negation-on-z-well-defined}
Negation determines a well-defined unary operation $\mathbb{Z}\to\mathbb{Z}$.
\end{theorem}
\end{theorembox}

\begin{theorembox}{Theorem}
\begin{theorem}[Negation Gives a Left Additive Inverse on $\mathbb{Z}$]
\label{thm:negation-left-additive-inverse-on-z}
For every $x\in\mathbb{Z}$,
\[
(-x)+x=0_{\mathbb{Z}}.
\]
\end{theorem}
\end{theorembox}

\begin{corollarybox}{Corollary}
\begin{corollary}[Negation Gives a Right Additive Inverse on $\mathbb{Z}$]
\label{cor:negation-right-additive-inverse-on-z}
For every $x\in\mathbb{Z}$,
\[
x+(-x)=0_{\mathbb{Z}}.
\]
\end{corollary}
\end{corollarybox}

\begin{theorembox}{Theorem}
\begin{theorem}[Every Integer Has an Additive Inverse]
\label{thm:every-integer-has-additive-inverse}
For every $x\in\mathbb{Z}$ there exists $y\in\mathbb{Z}$ such that
\[
x+y=0_{\mathbb{Z}}
\qquad\text{and}\qquad
y+x=0_{\mathbb{Z}}.
\]
\end{theorem}
\end{theorembox}

\subsection*{Distributivity on $\mathbb{Z}$-Classes}

\begin{theorembox}{Theorem}
\begin{theorem}[Left Distributivity on $\mathbb{Z}$]
\label{thm:left-distributivity-on-z}
For all $x,y,z\in\mathbb{Z}$,
\[
x\cdot(y+z)=x\cdot y+x\cdot z.
\]
\end{theorem}
\end{theorembox}

\begin{corollarybox}{Corollary}
\begin{corollary}[Right Distributivity on $\mathbb{Z}$]
\label{cor:right-distributivity-on-z}
For all $x,y,z\in\mathbb{Z}$,
\[
(y+z)\cdot x=y\cdot x+z\cdot x.
\]
\end{corollary}
\end{corollarybox}

\begin{theorembox}{Theorem}
\begin{theorem}[Multiplication Distributes over Addition on $\mathbb{Z}$]
\label{thm:multiplication-distributes-over-addition-on-z}
Multiplication distributes over addition on both sides in $\mathbb{Z}$.
\end{theorem}
\end{theorembox}

\subsection*{Class-Level Ring Summary}

\begin{theorembox}{Theorem}
\begin{theorem}[$\mathbb{Z}$ Is an Additive Abelian Group]
\label{thm:z-additive-abelian-group}
The structure $(\mathbb{Z},+,0_{\mathbb{Z}})$ is an abelian group.
\end{theorem}
\end{theorembox}

\begin{theorembox}{Theorem}
\begin{theorem}[$\mathbb{Z}$ Is a Multiplicative Commutative Monoid]
\label{thm:z-multiplicative-commutative-monoid}
The structure $(\mathbb{Z},\cdot,1_{\mathbb{Z}})$ is a commutative monoid.
\end{theorem}
\end{theorembox}

\begin{theorembox}{Theorem}
\begin{theorem}[$\mathbb{Z}$ Is a Commutative Ring]
\label{thm:z-commutative-ring}
The structure $(\mathbb{Z},+,\cdot,0_{\mathbb{Z}},1_{\mathbb{Z}})$ is a
commutative ring.
\end{theorem}
\end{theorembox}
