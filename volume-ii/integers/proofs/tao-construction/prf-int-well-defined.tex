\begin{theorem*}[Tao 4.1.3 — Addition and multiplication are well-defined]
Let $a, b, a', b', c, d \in \mathbb{N}$.
If $(a {-\!\!-} b) = (a' {-\!\!-} b')$, then:
\begin{align*}
(a {-\!\!-} b) + (c {-\!\!-} d) &= (a' {-\!\!-} b') + (c {-\!\!-} d), \\
(a {-\!\!-} b) \times (c {-\!\!-} d) &= (a' {-\!\!-} b') \times (c {-\!\!-} d),
\end{align*}
and similarly when replacing $(c {-\!\!-} d)$ by an equal integer.
\end{theorem*}
