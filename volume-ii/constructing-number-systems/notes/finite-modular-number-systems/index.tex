\section{Finite Modular Number Systems}
\begin{toolkitbox}{Toolkit: Residue Class Arithmetic}
\begin{itemize}
  \item \textbf{Modulus:} a positive integer \(n\).
  \item \textbf{Congruence:} \(a\equiv b\pmod n\) when \(n\mid(a-b)\).
  \item \textbf{Elements:} equivalence classes of integers modulo \(n\).
  \item \textbf{Operations:} class addition and multiplication inherited from \(\mathbb{Z}\).
\end{itemize}
\end{toolkitbox}
