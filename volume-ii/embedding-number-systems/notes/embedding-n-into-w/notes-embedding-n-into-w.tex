\section{Embedding $\mathbb{N}$ into $\mathbb{W}$}
\label{sec:embedding-n-into-w}

\begin{topicbox}{Embedding $\mathbb{N}$ into $\mathbb{W}$}
\begin{exposition}
This section records, from the global embedding viewpoint, the local bridge
proved in the whole-number chapter: $\mathbb{N}$ sits inside $\mathbb{W}$ as
the positive part. The map is the bridge that lets elements of $\mathbb{N}$ be
regarded as nonzero elements of $\mathbb{W}$ after identification.
\end{exposition}
\end{topicbox}
\begin{definitionbox}{Definition (Embedding $\mathbb{N}$ into $\mathbb{W}$)}
\begin{definition}[Embedding $\mathbb{N}$ into $\mathbb{W}$]
\label{def:embedding-n-into-w}
The \emph{canonical embedding} $\iota:\mathbb{N}\to \mathbb{W}$ is defined by
\[
\iota(n)=n \quad(\text{the copy of each natural number inside the whole numbers}).
\]
\end{definition}
\end{definitionbox}
\begin{remark*}[Standard quantified statement]
\[
\iota:\mathbb{N}\to \mathbb{W},\qquad \iota(n)=n \quad(\text{the copy of each natural number inside the whole numbers}).
\]
\end{remark*}
\begin{remark*}[Interpretation]
The map sends each element of $\mathbb{N}$ to its natural representative in $\mathbb{W}$. The next result certifies that this representative choice preserves all arithmetic and order, so $\mathbb{N}$ may be identified with its image in $\mathbb{W}$.
\end{remark*}

\begin{theorembox}{Theorem ($\mathbb{N}$ Embeds in $\mathbb{W}$)}
\begin{theorem}[$\mathbb{N}$ Embeds in $\mathbb{W}$]
\label{thm:n-embeds-in-w}
The canonical map $\iota:\mathbb{N}\to \mathbb{W}$ identifies $\mathbb{N}$ with
the positive part $\mathbb{W}\setminus\{0\}$ and preserves successor, $1$,
addition, multiplication, and order.
\hyperref[prf:n-embeds-in-w]{Go to proof.}
\end{theorem}
\end{theorembox}
\begin{remark*}[Standard quantified statement]
\[
\iota:\mathbb{N}\to \mathbb{W}\setminus\{0\} \text{ is bijective and preserves successor, arithmetic, and order.}
\]
\end{remark*}
\begin{remark*}[Interpretation]
Once this theorem is in place, $\mathbb{N}$ may be treated as the positive
substructure $\iota(\mathbb{N})=\mathbb{W}\setminus\{0\}$ inside
$\mathbb{W}$. The new element of $\mathbb{W}$ is exactly the adjoined zero.
\end{remark*}
