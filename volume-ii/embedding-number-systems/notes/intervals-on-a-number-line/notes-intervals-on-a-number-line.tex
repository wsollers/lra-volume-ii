\section{Intervals on a Number Line}
\label{sec:intervals-on-a-number-line}

\begin{topicbox}{The Basic Pieces of the Line}
\begin{exposition}
Intervals are the natural connected pieces of a number line. This section gives each named bounded interval its own definition and then characterizes intervals intrinsically by order-convexity. That characterization is the hinge reused in the topology chapter, where intervals reappear as the connected subsets of $\mathbb{R}$.
\end{exposition}
\end{topicbox}
\begin{tcolorbox}[colback=defbox, colframe=defborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Definition (Open Interval)}},
  fonttitle=\small\bfseries]
\begin{definition}[Open Interval]
\label{def:open-interval}
Let $a,b\in\mathbb{R}$ with $a\le b$. The \emph{open interval} with endpoints $a$ and $b$ is
\[
(a,b)=\{\,x\in\mathbb{R} : a<x<b\,\}.
\]
\end{definition}
\end{tcolorbox}
\begin{remark*}[Standard quantified statement]
\[
(a,b)=\{\,x\in\mathbb{R} : a<x<b\,\}.
\]
\end{remark*}
\begin{remark*}[Interpretation]
The open interval excludes both endpoints. It is one of the four bounded interval shapes.
\end{remark*}

\begin{tcolorbox}[colback=defbox, colframe=defborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Definition (Ray)}},
  fonttitle=\small\bfseries]
\begin{definition}[Ray]
\label{def:ray}
Let $a\in\mathbb{R}$. The four \emph{rays} determined by $a$ are
\[
(a,\infty)=\{x:x>a\},\ [a,\infty)=\{x:x\ge a\},\ (-\infty,a)=\{x:x<a\},\ (-\infty,a]=\{x:x\le a\}.
\]
\end{definition}
\end{tcolorbox}
\begin{remark*}[Interpretation]
Rays are the unbounded one-sided intervals. Together with the bounded intervals and the whole line $(-\infty,\infty)=\mathbb{R}$, they exhaust the interval shapes.
\end{remark*}

\begin{tcolorbox}[colback=defbox, colframe=defborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Definition (Order-Convex Set)}},
  fonttitle=\small\bfseries]
\begin{definition}[Order-Convex Set]
\label{def:order-convex-set}
A set $I\subseteq\mathbb{R}$ is \emph{order-convex} exactly when
\[
\forall x,y\in I\;\forall z\in\mathbb{R}\;\bigl((x<z<y)\Longrightarrow z\in I\bigr).
\]
\end{definition}
\end{tcolorbox}
\begin{remark*}[Standard quantified statement]
\[
I \text{ is order-convex} \Longleftrightarrow \forall x,y\in I\;\forall z\in\mathbb{R}\;(x<z<y\Rightarrow z\in I).
\]
\end{remark*}
\begin{remark*}[Negated quantified statement]
\[
I \text{ is not order-convex} \Longleftrightarrow \exists x,y\in I\;\exists z\in\mathbb{R}\;(x<z<y\land z\notin I).
\]
\end{remark*}
\begin{remark*}[Interpretation]
Order-convexity says $I$ has no internal gaps: whenever two points lie in $I$, everything between them does too. This is the intrinsic property that the named interval shapes all share.
\end{remark*}

\begin{tcolorbox}[colback=thmbox, colframe=thmborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Theorem (Intervals Are Exactly the Order-Convex Sets)}},
  fonttitle=\small\bfseries]
\begin{theorem}[Intervals Are Exactly the Order-Convex Sets]
\label{thm:interval-iff-order-convex}
A nonempty set $I\subseteq\mathbb{R}$ is an interval (bounded interval, ray, or all of $\mathbb{R}$) if and only if $I$ is order-convex.
\hyperref[prf:interval-iff-order-convex]{Go to proof.}
\end{theorem}
\end{tcolorbox}
\begin{remark*}[Standard quantified statement]
\[
I \text{ is an interval} \Longleftrightarrow I\neq\emptyset \land \forall x,y\in I\;\forall z\in\mathbb{R}\;(x<z<y\Rightarrow z\in I).
\]
\end{remark*}
\begin{remark*}[Interpretation]
This theorem replaces the case-by-case list of interval shapes with a single intrinsic property. In the topology chapter it becomes the statement that the intervals are precisely the connected subsets of $\mathbb{R}$ --- a fact special to the line.
\end{remark*}
