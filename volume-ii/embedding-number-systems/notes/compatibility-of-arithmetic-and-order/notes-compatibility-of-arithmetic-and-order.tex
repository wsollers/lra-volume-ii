\label{sec:compatibility-of-arithmetic-and-order}

\begin{topicbox}{One Arithmetic Through the Chain}
\begin{exposition}
Each embedding preserves arithmetic and order individually. What remains is to confirm that the embeddings are mutually compatible: composing $\mathbb{N}\hookrightarrow\mathbb{W}\hookrightarrow\mathbb{Z}\hookrightarrow\mathbb{Q}\hookrightarrow\mathbb{R}$ agrees with the direct embedding of $\mathbb{N}$ into $\mathbb{R}$, so that ``$2$'' has one unambiguous meaning after all identifications.
\end{exposition}
\end{topicbox}
\begin{theorembox}{Theorem (Compatibility of the Embedding Chain)}
\begin{theorem}[Compatibility of the Embedding Chain]
\label{thm:embedding-chain-compatibility}
Let $\iota_{1}:\mathbb{N}\to\mathbb{W}$, $\iota_{2}:\mathbb{W}\to\mathbb{Z}$, $\iota_{3}:\mathbb{Z}\to\mathbb{Q}$, and $\iota_{4}:\mathbb{Q}\to\mathbb{R}$ be the canonical embeddings. For every $n\in\mathbb{N}$ the composite $(\iota_{4}\circ\iota_{3}\circ\iota_{2}\circ\iota_{1})(n)$ is the real number identified with $n$, and the composite preserves the arithmetic and order carried by the source system. Hence arithmetic computed in any system agrees with arithmetic on its image in $\mathbb{R}$.
\hyperref[prf:embedding-chain-compatibility]{Go to proof.}
\end{theorem}
\end{theorembox}
\begin{remark*}[Standard quantified statement]
\[
\forall n,m\in\mathbb{N}\;\bigl(\Phi(n+m)=\Phi(n)+\Phi(m)\ \land\ \Phi(n\cdot m)=\Phi(n)\cdot\Phi(m)\ \land\ (n<m\Leftrightarrow\Phi(n)<\Phi(m))\bigr),\quad \Phi=\iota_{4}\circ\iota_{3}\circ\iota_{2}\circ\iota_{1}.
\]
\end{remark*}
\begin{remark*}[Interpretation]
Compatibility is what makes the single symbol $2$ well-defined across the whole chain: the answer to $2+2$ is the same whether computed in $\mathbb{N}$ or after embedding into $\mathbb{R}$. Without this coherence the inclusion notation would silently carry two different meanings for the same numeral.
\end{remark*}
