\section{Constructing a Number System}
\begin{toolkitbox}{Toolkit: Construction Obligations}
\begin{itemize}
  \item \textbf{Carrier:} the underlying collection of objects.
  \item \textbf{Operations:} the arithmetic rules defined on the carrier.
  \item \textbf{Relations:} equality, equivalence, order, or congruence relations.
  \item \textbf{Verification:} closure, well-definedness, laws, and embeddings.
\end{itemize}
\end{toolkitbox}
