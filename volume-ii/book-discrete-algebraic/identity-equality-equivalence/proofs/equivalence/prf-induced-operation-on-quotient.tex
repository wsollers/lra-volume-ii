\begin{theorem*}[Induced Operation on a Quotient]
If $O:A\times A\to A$ respects $\sim$, then
\[
[a]\mathbin{\overline{O}}[b]:=[a\mathbin{O}b]
\]
defines a well-defined binary operation on $A/{\sim}$.
\end{theorem*}
