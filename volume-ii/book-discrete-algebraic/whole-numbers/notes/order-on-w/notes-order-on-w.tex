\subsection*{The Extended Order}

\begin{definitionbox}{Definition (Order on \texorpdfstring{$\mathbb{W}$}{W})}
\begin{definition}[Order on $\mathbb{W}$]
\label{def:order-on-w}
Define $\leq_{\mathbb{W}}$ on $\mathbb{W}$ by declaring
\[
0\leq_{\mathbb{W}}a\quad(a\in\mathbb{W}),
\]
and, for $a,b\in\mathbb{N}$,
\[
a\leq_{\mathbb{W}}b
\quad\Longleftrightarrow\quad
a\leq_{\mathbb{N}}b.
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
\forall a,b\in\mathbb{W}\;
\bigl(a\leq_{\mathbb{W}}b
\Longleftrightarrow
(a=0\ \vee\ (a,b\in\mathbb{N}\ \wedge\ a\leq_{\mathbb{N}}b))\bigr).
\]
\end{remark*}

\begin{remark*}[Interpretation]
The non-strict order makes $0$ least and otherwise preserves the order already
defined on $\mathbb{N}$.
\end{remark*}

\begin{definitionbox}{Definition (Strict Order on \texorpdfstring{$\mathbb{W}$}{W})}
\begin{definition}[Strict Order on $\mathbb{W}$]
\label{def:strict-order-on-w}
Define $<_{\mathbb{W}}$ on $\mathbb{W}$ by declaring
\[
0<_{\mathbb{W}}n\quad(n\in\mathbb{N}),
\]
and, for $a,b\in\mathbb{N}$,
\[
a<_{\mathbb{W}}b
\quad\Longleftrightarrow\quad
a<_{\mathbb{N}}b.
\]
No element of $\mathbb{W}$ is strictly less than $0$.
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
\[
\forall a,b\in\mathbb{W}\;
\bigl(a<_{\mathbb{W}}b
\Longleftrightarrow
(a=0\ \wedge\ b\in\mathbb{N})
\vee
(a,b\in\mathbb{N}\ \wedge\ a<_{\mathbb{N}}b)\bigr).
\]
\end{remark*}

\begin{remark*}[Interpretation]
The non-strict order makes $0$ least. The strict order says exactly that the
positive natural numbers lie after $0$ and keep their old strict ordering.
\end{remark*}

\begin{theorembox}{Theorem (Order on \texorpdfstring{$\mathbb{W}$}{W} Extends Order on \texorpdfstring{$\mathbb{N}$}{N})}
\begin{theorem}[Order on $\mathbb{W}$ Extends Order on $\mathbb{N}$]
\label{thm:order-on-w-extends-order-on-n}
For all $a,b\in\mathbb{N}$,
\[
a\leq_{\mathbb{W}}b\Longleftrightarrow a\leq_{\mathbb{N}}b,
\qquad
a<_{\mathbb{W}}b\Longleftrightarrow a<_{\mathbb{N}}b.
\]
\hyperref[prf:order-on-w-extends-order-on-n]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\forall a,b\in\mathbb{N}\;
\bigl(
(a\leq_{\mathbb{W}}b\Longleftrightarrow a\leq_{\mathbb{N}}b)
\wedge
(a<_{\mathbb{W}}b\Longleftrightarrow a<_{\mathbb{N}}b)
\bigr).
\]
\end{remark*}

\begin{remark*}[Interpretation]
The extended order changes only comparisons involving the new element $0$.
\end{remark*}

\begin{theorembox}{Theorem (Zero Is the Least Element of \texorpdfstring{$\mathbb{W}$}{W})}
\begin{theorem}[Zero Is the Least Element of $\mathbb{W}$]
\label{thm:zero-least-element-of-w}
For every $a\in\mathbb{W}$,
\[
0\leq_{\mathbb{W}}a.
\]
\hyperref[prf:zero-least-element-of-w]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\forall a\in\mathbb{W}\;(0\leq_{\mathbb{W}}a).
\]
\end{remark*}

\begin{remark*}[Interpretation]
The adjoined element $0$ is the first element of the whole-number order.
\end{remark*}

\begin{theorembox}{Theorem (Order on \texorpdfstring{$\mathbb{W}$}{W} Is Reflexive)}
\begin{theorem}[Order on $\mathbb{W}$ Is Reflexive]
\label{thm:order-on-w-reflexive}
For every $a\in\mathbb{W}$, $a\leq_{\mathbb{W}}a$.
\hyperref[prf:order-on-w-reflexive]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\forall a\in\mathbb{W}\;(a\leq_{\mathbb{W}}a).
\]
\end{remark*}

\begin{remark*}[Interpretation]
Reflexivity is immediate for $0$ and inherited from $\mathbb{N}$ for positive
elements.
\end{remark*}

\begin{theorembox}{Theorem (Order on \texorpdfstring{$\mathbb{W}$}{W} Is Antisymmetric)}
\begin{theorem}[Order on $\mathbb{W}$ Is Antisymmetric]
\label{thm:order-on-w-antisymmetric}
For all $a,b\in\mathbb{W}$,
\[
a\leq_{\mathbb{W}}b\ \text{and}\ b\leq_{\mathbb{W}}a
\Longrightarrow a=b.
\]
\hyperref[prf:order-on-w-antisymmetric]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\forall a,b\in\mathbb{W}\;
\bigl(a\leq_{\mathbb{W}}b\ \wedge\ b\leq_{\mathbb{W}}a
\Longrightarrow a=b\bigr).
\]
\end{remark*}

\begin{remark*}[Interpretation]
Antisymmetry follows from the least-element behavior of $0$ and the inherited
antisymmetry on positive natural numbers.
\end{remark*}

\begin{theorembox}{Theorem (Order on \texorpdfstring{$\mathbb{W}$}{W} Is Transitive)}
\begin{theorem}[Order on $\mathbb{W}$ Is Transitive]
\label{thm:order-on-w-transitive}
For all $a,b,c\in\mathbb{W}$,
\[
a\leq_{\mathbb{W}}b\ \text{and}\ b\leq_{\mathbb{W}}c
\Longrightarrow a\leq_{\mathbb{W}}c.
\]
\hyperref[prf:order-on-w-transitive]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\forall a,b,c\in\mathbb{W}\;
\bigl(a\leq_{\mathbb{W}}b\ \wedge\ b\leq_{\mathbb{W}}c
\Longrightarrow a\leq_{\mathbb{W}}c\bigr).
\]
\end{remark*}

\begin{remark*}[Interpretation]
Transitivity reduces to the least-element clause when $0$ occurs and to the
transitivity theorem on $\mathbb{N}$ otherwise.
\end{remark*}

\begin{theorembox}{Theorem (Order on \texorpdfstring{$\mathbb{W}$}{W} Is Total)}
\begin{theorem}[Order on $\mathbb{W}$ Is Total]
\label{thm:order-on-w-total}
For all $a,b\in\mathbb{W}$,
\[
a\leq_{\mathbb{W}}b
\quad\text{or}\quad
b\leq_{\mathbb{W}}a.
\]
\hyperref[prf:order-on-w-total]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\forall a,b\in\mathbb{W}\;
\bigl(a\leq_{\mathbb{W}}b\ \vee\ b\leq_{\mathbb{W}}a\bigr).
\]
\end{remark*}

\begin{remark*}[Interpretation]
Any comparison involving $0$ is decided by leastness, while comparisons
between positive elements are decided by totality on $\mathbb{N}$.
\end{remark*}

\begin{theorembox}{Theorem (Well-Ordering Principle for \texorpdfstring{$\mathbb{W}$}{W})}
\begin{theorem}[Well-Ordering Principle for $\mathbb{W}$]
\label{thm:well-ordering-principle-for-w}
Every nonempty subset of $\mathbb{W}$ has a least element with respect to
$\leq_{\mathbb{W}}$.
\hyperref[prf:well-ordering-principle-for-w]{Go to proof.}
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
\[
\forall A\subseteq\mathbb{W}\;
\bigl(A\neq\varnothing \Longrightarrow
\exists m\in A\;\forall a\in A\;(m\leq_{\mathbb{W}}a)\bigr).
\]
\end{remark*}

\begin{remark*}[Interpretation]
A nonempty subset either contains $0$, which is least, or lies in the positive
part where the well-ordering principle for $\mathbb{N}$ applies.
\end{remark*}
