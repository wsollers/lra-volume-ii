\subsection*{Congruence and Residue Classes}

\begin{remark*}[Exposition]
Modular arithmetic is the quotient version of the two-element example. The
carrier is obtained by identifying integers that have the same remainder
modulo a fixed positive integer.
\end{remark*}

\begin{definitionbox}{Definition}
\begin{definition}[Congruence Modulo an Integer]
\label{def:congruence-modulo-integer}
Let \(n\in\mathbb{Z}\) be positive. For \(a,b\in\mathbb{Z}\), define
\[
a\equiv b \pmod n
\quad:\Longleftrightarrow\quad
n\mid (a-b).
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
For positive integer \(n\), two integers are congruent modulo \(n\) exactly
when \(n\) divides their difference.
\end{remark*}

\begin{remark*}[Interpretation]
Congruence modulo \(n\) records sameness of remainder using divisibility in
\(\mathbb{Z}\).
\end{remark*}

\begin{theorembox}{Theorem}
\begin{theorem}[Congruence Modulo an Integer Is an Equivalence Relation]
\label{thm:congruence-modulo-integer-equivalence}
For each positive integer \(n\), congruence modulo \(n\) is an equivalence
relation on \(\mathbb{Z}\).
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
For each positive integer \(n\), congruence modulo \(n\) is reflexive,
symmetric, and transitive on \(\mathbb{Z}\).
\end{remark*}

\begin{remark*}[Interpretation]
This permits the quotient set \(\mathbb{Z}/n\mathbb{Z}\) to be formed from
congruence classes.
\end{remark*}

\begin{definitionbox}{Definition}
\begin{definition}[Integers Modulo \(n\)]
\label{def:integers-modulo-n}
Let \(n\in\mathbb{Z}\) be positive. The \emph{integers modulo \(n\)} are the
quotient set
\[
\mathbb{Z}/n\mathbb{Z}:=\mathbb{Z}/{\equiv \pmod n}.
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
For positive integer \(n\), \(\mathbb{Z}/n\mathbb{Z}\) is the quotient of
\(\mathbb{Z}\) by congruence modulo \(n\).
\end{remark*}

\begin{remark*}[Interpretation]
The elements are not single integers but equivalence classes of integers.
\end{remark*}

\begin{definitionbox}{Definition}
\begin{definition}[Residue Class Modulo \(n\)]
\label{def:residue-class-modulo-n}
For \(a\in\mathbb{Z}\), the \emph{residue class of \(a\) modulo \(n\)} is
\[
[a]_n:=\{b\in\mathbb{Z}: b\equiv a\pmod n\}.
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
The set \([a]_n\) contains exactly the integers congruent to \(a\) modulo \(n\).
\end{remark*}

\begin{remark*}[Interpretation]
Each such set is one element of the quotient number system.
\end{remark*}

\subsection*{Operations}

\begin{definitionbox}{Definition}
\begin{definition}[Addition Modulo \(n\)]
\label{def:addition-modulo-n}
For residue classes modulo a positive integer \(n\), define
\[
[a]_n+[b]_n:=[a+b]_n.
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
The sum \([a]_n+[b]_n\) is \([a+b]_n\).
\end{remark*}

\begin{remark*}[Interpretation]
This formula becomes a valid operation on classes only after well-definedness
is proved.
\end{remark*}

\begin{definitionbox}{Definition}
\begin{definition}[Multiplication Modulo \(n\)]
\label{def:multiplication-modulo-n}
For residue classes modulo a positive integer \(n\), define
\[
[a]_n\cdot[b]_n:=[ab]_n.
\]
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
The product \([a]_n\cdot[b]_n\) is \([ab]_n\).
\end{remark*}

\begin{remark*}[Interpretation]
This formula inherits multiplication from \(\mathbb{Z}\), subject to the same
well-definedness obligation.
\end{remark*}

\begin{theorembox}{Theorem}
\begin{theorem}[Modular Addition and Multiplication Are Well-Defined]
\label{thm:modular-operations-well-defined}
For each positive integer \(n\), addition and multiplication modulo \(n\) are
well-defined operations on \(\mathbb{Z}/n\mathbb{Z}\).
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
For each positive integer \(n\), the representative formulas for modular
addition and multiplication determine unique outputs.
\end{remark*}

\begin{remark*}[Interpretation]
Changing representatives does not change the output, so the operations
belong to \(\mathbb{Z}/n\mathbb{Z}\) itself.
\end{remark*}

\begin{definitionbox}{Definition}
\begin{definition}[Finite Modular Number System]
\label{def:finite-modular-number-system}
For a positive integer \(n\), the \emph{finite modular number system modulo
\(n\)} is
\[
\mathbb{Z}/n\mathbb{Z}
\]
with addition and multiplication modulo \(n\), additive identity \([0]_n\),
and multiplicative identity \([1]_n\).
\end{definition}
\end{definitionbox}

\begin{remark*}[Standard quantified statement]
For positive integer \(n\), the finite modular number system modulo \(n\) is
\(\mathbb{Z}/n\mathbb{Z}\) with modular addition, modular multiplication,
\([0]_n\), and \([1]_n\).
\end{remark*}

\begin{remark*}[Interpretation]
This is the quotient arithmetic system obtained by reducing integer
arithmetic modulo \(n\).
\end{remark*}

\subsection*{Basic Example}

\begin{theorembox}{Theorem}
\begin{theorem}[\(\mathbb{Z}/4\mathbb{Z}\) Has Zero Divisors]
\label{thm:z-mod-four-has-zero-divisors}
In \(\mathbb{Z}/4\mathbb{Z}\), the nonzero class \([2]_4\) satisfies
\[
[2]_4\cdot[2]_4=[0]_4.
\]
Therefore \(\mathbb{Z}/4\mathbb{Z}\) is not a field.
\end{theorem}
\end{theorembox}

\begin{remark*}[Standard quantified statement]
In \(\mathbb{Z}/4\mathbb{Z}\), \([2]_4\neq[0]_4\) and
\([2]_4\cdot[2]_4=[0]_4\).
\end{remark*}

\begin{remark*}[Interpretation]
This shows that finite modular number systems need not be fields.
\end{remark*}
