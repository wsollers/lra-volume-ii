\begin{theorem*}[Additive Cancellation]
Let $F$ be a field and let $a,b,c\in F$.  If
\[
a + c = b + c,
\]
then $a = b$.
\end{theorem*}

\begin{proof}
\textbf{Professional Standard Proof.}~\\
Assume $a + c = b + c$.  Adding $-c$ to both sides and applying
associativity and the additive inverse and identity axioms:
\[
a = a + 0 = a + (c + (-c)) = (a + c) + (-c) = (b + c) + (-c) = b + (c + (-c)) = b + 0 = b.
\]
\end{proof}

\begin{proof}
\textbf{Detailed Learning Proof.}~\\

\textbf{Step 1.} State the hypothesis.

Assume $a + c = b + c$.

\textbf{Step 2.} Add $-c$ to both sides.

Since $-c\in F$ (additive inverses exist in $F$), add $-c$ to both sides:
\[
(a + c) + (-c) = (b + c) + (-c).
\]

\textbf{Step 3.} Apply associativity of addition to both sides.

By associativity,
\[
(a + c) + (-c) = a + (c + (-c))
\qquad\text{and}\qquad
(b + c) + (-c) = b + (c + (-c)).
\]

\textbf{Step 4.} Apply the additive inverse axiom.

By the additive inverse axiom, $c + (-c) = 0$.  Therefore:
\[
a + (c + (-c)) = a + 0
\qquad\text{and}\qquad
b + (c + (-c)) = b + 0.
\]

\textbf{Step 5.} Apply the additive identity axiom.

By the additive identity axiom, $a + 0 = a$ and $b + 0 = b$.

\textbf{Step 6.} Conclude.

Chaining Steps 2--5:
\[
a = a + 0 = a + (c + (-c)) = (a+c) + (-c) = (b+c) + (-c) = b + (c+(-c)) = b + 0 = b.
\]
Therefore $a = b$.
\end{proof}

\begin{remark*}[Proof structure]
The proof adds the additive inverse $-c$ to both sides of the hypothesis,
then unwinds the result using associativity, the additive inverse axiom, and
the additive identity axiom.  Each step cites exactly one field axiom.  The
conclusion $a = b$ is reached by a chain of equalities with no auxiliary
lemmas required.
\end{remark*}
