\begin{theorem*}[Scaling Numerator and Denominator by the Same Nonzero Rational]
For every $a,b,e\in\mathbb{Q}$ with $b\neq 0$ and $e\neq 0$,
\[
\frac{a}{b}=\frac{a\cdot e}{b\cdot e}.
\]
\end{theorem*}

\begin{proof}
\textbf{Professional Standard Proof.}~\\
Let $a,b,e\in\mathbb{Q}$ with
\[
b\neq 0
\qquad\text{and}\qquad
e\neq 0.
\]
Since $\mathbb{Q}$ is an integral domain, the product of two nonzero
rational numbers is nonzero. Hence
\[
b\cdot e\neq 0.
\]
By the cross-multiplication criterion, it is enough to prove
\[
a\cdot (b\cdot e)=b\cdot (a\cdot e).
\]
Using associativity and commutativity of multiplication in $\mathbb{Q}$,
\[
a\cdot (b\cdot e)
=
(a\cdot b)\cdot e
=
(b\cdot a)\cdot e
=
b\cdot (a\cdot e).
\]
Therefore
\[
\frac{a}{b}=\frac{a\cdot e}{b\cdot e}.
\]
\end{proof}

\begin{proof}
\textbf{Detailed Learning Proof.}~\\

\textbf{Step 1.} Check that the scaled denominator is nonzero.

Let $a,b,e\in\mathbb{Q}$ with
\[
b\neq 0
\qquad\text{and}\qquad
e\neq 0.
\]
Since $\mathbb{Q}$ is an integral domain, the product of two nonzero
elements is nonzero. Therefore
\[
b\cdot e\neq 0.
\]
Thus both quotients
\[
\frac{a}{b}
\qquad\text{and}\qquad
\frac{a\cdot e}{b\cdot e}
\]
are defined.

\textbf{Step 2.} Reduce the desired equality to a cross-product equality.

By the cross-multiplication criterion, the equality
\[
\frac{a}{b}=\frac{a\cdot e}{b\cdot e}
\]
is equivalent to
\[
a\cdot (b\cdot e)=b\cdot (a\cdot e).
\]
Hence it remains to prove this product equality in $\mathbb{Q}$.

\textbf{Step 3.} Prove the cross-product equality.

Using associativity and commutativity of multiplication in $\mathbb{Q}$,
\[
a\cdot (b\cdot e)
=
(a\cdot b)\cdot e
=
(b\cdot a)\cdot e
=
b\cdot (a\cdot e).
\]
Therefore
\[
a\cdot (b\cdot e)=b\cdot (a\cdot e).
\]

\textbf{Step 4.} Conclude the quotient equality.

Since
\[
a\cdot (b\cdot e)=b\cdot (a\cdot e),
\]
the cross-multiplication criterion gives
\[
\frac{a}{b}=\frac{a\cdot e}{b\cdot e}.
\]
This proves the theorem.
\end{proof}

\begin{remark*}[Proof structure]
The proof is direct. First, the scaled denominator $b\cdot e$ is shown
to be nonzero. Then the desired equality of quotients is reduced by the
cross-multiplication criterion to an equality of products. That product
equality follows from associativity and commutativity of multiplication
in $\mathbb{Q}$.
\end{remark*}
