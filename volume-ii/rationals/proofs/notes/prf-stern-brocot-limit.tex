\begin{theorem*}[Stern--Brocot Limit Theorem]
Every irrational $x > 0$ determines a unique infinite path in the
Stern--Brocot tree, and the endpoint sequences converge to $x$.
Conversely, every infinite path determines a unique positive irrational
as its limit.
\end{theorem*}

\begin{proof}[Proof (Professional Standard)]
\textbf{Forward direction (irrational $\Rightarrow$ unique infinite path
converging to $x$).}

By \hyperref[thm:stern-brocot-refinement]{Theorem~\ref*{thm:stern-brocot-refinement}},
the SB refinement process generates a nested sequence of rational intervals
$I_n$ each containing $x$.  Since $x$ is irrational, it never equals any
mediant, so the process never terminates: the path is infinite.

The left-right decisions at each step are deterministic (determined by the
sign of $x - m_n$), so the path is unique.

By \hyperref[cor:sb-path-limit]{Corollary~\ref*{cor:sb-path-limit}},
the endpoint sequences converge to $x$.

\textbf{Converse direction (infinite path $\Rightarrow$ unique irrational limit).}

By \hyperref[prop:sb-unique-point]{Proposition~\ref*{prop:sb-unique-point}},
the nested intervals $(I_n)$ associated to any infinite path contain a
unique point $\xi \in \mathbb{R}$.

By \hyperref[cor:sb-limit-irrational]{Corollary~\ref*{cor:sb-limit-irrational}},
an infinite path cannot terminate at a rational, so $\xi \notin \mathbb{Q}$.
Therefore $\xi$ is irrational.

Uniqueness of $\xi$ follows from the Unique Point Proposition.
\end{proof}

\begin{proof}[Proof (Detailed Learning)]
\textbf{Forward direction.}

\textbf{Step 1: The path is infinite.}
Since $x$ is irrational, $x \neq (a_n + c_n)/(b_n + d_n)$ at every
step (the mediant is always rational).  Therefore the comparison $x$ vs.\ 
$m_n$ always yields a strict inequality, and the process never terminates.

\textbf{Step 2: The path is unique.}
At each step, the decision (go left or go right) is determined by the
strict comparison $x < m_n$ or $x > m_n$, which has a unique outcome
since $x$ is irrational and $m_n$ is rational.  Therefore the sequence
of left-right choices is unique, and the path is unique.

\textbf{Step 3: The path converges to $x$.}
By \hyperref[thm:stern-brocot-refinement]{Theorem~\ref*{thm:stern-brocot-refinement}},
$x \in I_n$ for all $n$.  By
\hyperref[cor:sb-path-limit]{Corollary~\ref*{cor:sb-path-limit}},
the unique point in $\bigcap I_n$ is $x$, and the endpoint sequences
converge to $x$.

\textbf{Converse direction.}

\textbf{Step 4: An infinite path has a unique real limit.}
By \hyperref[prop:sb-unique-point]{Proposition~\ref*{prop:sb-unique-point}},
the nested intervals of any infinite path contain exactly one point
$\xi \in \mathbb{R}$.

\textbf{Step 5: The limit is irrational.}
By \hyperref[cor:sb-limit-irrational]{Corollary~\ref*{cor:sb-limit-irrational}},
an infinite path corresponds to a non-rational target.  Since $\xi$ is
the unique point in $\bigcap I_n$, and the path is infinite (so the
process never encountered $\xi$ as a mediant), $\xi$ cannot be rational.
Therefore $\xi$ is irrational.

\textbf{Step 6: Conclude.}
Every irrational $x > 0$ determines a unique infinite path converging to
$x$; every infinite path determines a unique positive irrational as its
limit.
\end{proof}

\begin{remark*}[Proof structure]
The theorem is a synthesis of all the preceding Stern--Brocot results.
The forward direction uses the Refinement Theorem and Path Convergence
Corollary.  The converse uses the Unique Point Proposition and the
Irrational Limits Corollary.  No new ideas appear; the proof is an
assembly of the chain built up by Lemmas A and B, Proposition A, and
Corollaries A and B.
\end{remark*}
