\begin{theorem*}[Upper/Lower Perturbation Lemma, Rational Version]
If $q\in\mathbb{Q}$ and $\varepsilon\in\mathbb{Q}$ with $\varepsilon>0$, then there exist $r,s\in\mathbb{Q}$
such that
\[
q-\varepsilon < r < q < s < q+\varepsilon.
\]
\end{theorem*}

\begin{proof}
\textbf{Professional Standard Proof.}~\\
Let $q\in\mathbb{Q}$ and let $\varepsilon\in\mathbb{Q}$ with
\[
\varepsilon>0.
\]
By
\hyperref[thm:q-has-no-minimum-positive-element]{Theorem~\ref*{thm:q-has-no-minimum-positive-element}},
there exists $\delta\in\mathbb{Q}$ such that
\[
0<\delta<\varepsilon.
\]
Define
\[
r:=q-\delta
\qquad\text{and}\qquad
s:=q+\delta.
\]
Since $\mathbb{Q}$ is closed under addition and subtraction, we have
\[
r,s\in\mathbb{Q}.
\]

Because $\delta>0$, we have
\[
q-\delta<q<q+\delta,
\]
that is,
\[
r<q<s.
\]
And because $\delta<\varepsilon$, we have
\[
q-\varepsilon<q-\delta=r
\qquad\text{and}\qquad
s=q+\delta<q+\varepsilon.
\]
Combining these inequalities yields
\[
q-\varepsilon<r<q<s<q+\varepsilon.
\]
Thus there exist rational numbers $r,s$ with the required properties.
\end{proof}

\begin{remark*}[Proof structure]
Choose a smaller positive rational $\delta$ inside the interval $(0,\varepsilon)$.
Then place one rational point $\delta$ below $q$ and one rational point
$\delta$ above $q$.  Since $\delta<\varepsilon$, both perturbed points stay
within the larger window $(q-\varepsilon,q+\varepsilon)$, and since
$\delta>0$, they lie strictly on opposite sides of $q$.
\end{remark*}
