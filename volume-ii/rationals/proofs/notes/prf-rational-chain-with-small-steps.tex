\begin{theorem*}[Rational Chain with Uniformly Small Steps]
Let $r,s\in\mathbb{Q}$ with $r<s$, and let $p\in\mathbb{Q}$ with $p>0$.
Then there exist $m\in\mathbb{N}$ and rational numbers
\[
q_1,q_2,\dots,q_m
\]
such that
\[
q_1=r,
\qquad
q_m=s,
\]
and
\[
0<q_k-q_{k-1}<p
\qquad
\text{for every }k=2,\dots,m.
\]
\end{theorem*}
