\begin{theorem*}[Dirichlet's Approximation Theorem]
For every real number $x$ there exist infinitely many rational numbers $p/q$
with $p\in\mathbb{Z}$ and $q\in\mathbb{N}$ such that
\[
\left|x-\frac{p}{q}\right|<\frac{1}{q^2}.
\]
\end{theorem*}
