\begin{theorem*}[Rational Square Approximation from Above]
Let $r\in\mathbb{Q}$ with $r>0$, and let $n\in\mathbb{N}$ with $n\ge 1$.
Then there exists $s\in\mathbb{Q}$ with $s>0$ such that
\[
r<s^2<r+\frac1n.
\]
\end{theorem*}
