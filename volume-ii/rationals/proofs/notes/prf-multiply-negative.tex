\begin{theorem*}[Multiplication by a Negative Element Reverses Order]
If $a<b$ and $c<0$, then
\[
ac>bc.
\]
\end{theorem*}

\begin{proof}
\textbf{Professional Standard Proof.}~\\
Let $F$ be an ordered field, and suppose that
\[
a<b
\qquad\text{and}\qquad
c<0.
\]
Adding $-c$ to both sides of the inequality $c<0$, we obtain
\[
0<-c.
\]
Thus $-c>0$.

By \hyperref[thm:multiply-positive]{Theorem~\ref*{thm:multiply-positive}},
since $a<b$ and $-c>0$, we have
\[
a(-c)<b(-c).
\]
Using the negation-of-product identity,
\[
a(-c)=-(ac)
\qquad\text{and}\qquad
b(-c)=-(bc).
\]
Therefore
\[
-(ac)<-(bc).
\]
Add $bc$ to both sides:
\[
bc-(ac)<0.
\]
Since
\[
bc-(ac)=bc+(-(ac))=bc+(-ac)=bc-ac,
\]
adding $ac$ to both sides gives
\[
bc<ac.
\]
Equivalently,
\[
ac>bc.
\]
This is exactly the desired conclusion.
\end{proof}

\begin{remark*}[Proof structure]
The proof converts the negative multiplier into a positive one by replacing
$c$ with $-c$.  Since $-c>0$, the positive-multiplier theorem applies and
gives $a(-c)<b(-c)$.  The algebraic identity for negation of a product turns
this into $-(ac)<-(bc)$.  A final pair of additive translations removes the
leading minus signs and reverses the order into the desired form $ac>bc$.
\end{remark*}
