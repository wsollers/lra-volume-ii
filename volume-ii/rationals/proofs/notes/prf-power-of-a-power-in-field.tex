\begin{theorem*}[Power of a Power in a Field]
Let $(F,+,\cdot,0,1)$ be a field, let $q\in F\setminus\{0\}$, and let
$a,b\in\mathbb{Z}$. Then
\[
(q^a)^b=q^{ab}.
\]
\end{theorem*}

\begin{proof}
\textbf{Professional Standard Proof.}~\\
We split according to the sign of $b$.

If $b=0$, then
\[
(q^a)^0=1=q^0=q^{a\cdot 0}.
\]

If $b=n>0$, we argue by induction on $n$.  For $n=1$ the identity is
\[
(q^a)^1=q^a=q^{a\cdot 1}.
\]
Assume $(q^a)^n=q^{an}$.  Then
\[
(q^a)^{n+1}=(q^a)^n q^a=q^{an}q^a=q^{an+a}=q^{a(n+1)},
\]
using the induction hypothesis and the integer exponent addition law.

Finally, let $b=-n$ with $n>0$.  Then
\[
(q^a)^b=(q^a)^{-n}=((q^a)^{-1})^n=(q^{-a})^n=q^{(-a)n}=q^{a(-n)}=q^{ab},
\]
using the theorem on negative exponents and the already established
positive-exponent case.
\end{proof}

\begin{remark*}[Proof structure]
The proof separates the positive, zero, and negative cases for the outer
exponent $b$.  The positive case is an induction on $b$, with the induction
step supplied by the exponent addition law.  The negative case rewrites the
outer negative exponent as a positive power of an inverse and then applies
the positive case to $q^{-a}$.  This reduces the entire statement to the two
basic mechanisms already built into the integer exponent calculus: recursion
for positive powers and inversion for negative powers.
\end{remark*}
