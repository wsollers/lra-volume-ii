\subsection*{Prefractions}

\begin{definitionbox}{Definition}
\begin{definition}[Nonzero Integers]
\label{def:nonzero-integers}
The set of nonzero integers is
\[
\mathbb{Z}^{\ast}:=\mathbb{Z}\setminus\{0_{\mathbb{Z}}\}.
\]
\end{definition}
\end{definitionbox}

\begin{definitionbox}{Definition}
\begin{definition}[Prefraction]
\label{def:prefraction}
A \emph{prefraction} is an ordered pair $(a,b)\in\mathbb{Z}\times\mathbb{Z}^{\ast}$,
read as the formal quotient $a/b$.
\end{definition}
\end{definitionbox}

\subsection*{Fraction Equivalence}

\begin{definitionbox}{Definition}
\begin{definition}[Fraction Equivalence]
\label{def:fraction-equivalence}
For $(a,b),(c,d)\in\operatorname{Pre}\mathbb{Q}$, define
\[
(a,b)\sim_{\mathbb{Q}}(c,d)
\quad:\Longleftrightarrow\quad
a\cdot d=b\cdot c.
\]
\end{definition}
\end{definitionbox}

\begin{lemmabox}{Lemma}
\begin{lemma}[Fraction Equivalence Is Transitive]
\label{lem:fraction-equivalence-transitive}
For all prefractions $(a,b),(c,d),(e,f)$,

\subsection*{Definition of $\mathbb{Q}$}

\begin{definitionbox}{Definition}
\begin{definition}[Rational Numbers]
\label{def:rationals}
The rational numbers are the quotient
\[
\mathbb{Q}:=\operatorname{Pre}\mathbb{Q}/{\sim_{\mathbb{Q}}}.
\]
\end{definition}
\end{definitionbox}

\begin{dependencies}
\begin{itemize}
  \item \hyperref[def:quotient-set]{Quotient Set}
  \item \hyperref[def:prefraction-set]{Prefraction Set}
  \item \hyperref[def:fraction-equivalence]{Fraction Equivalence}
\end{itemize}
\end{dependencies}

\begin{definitionbox}{Definition}
\begin{definition}[Rational Class Notation]
\label{def:rational-class-notation}
The notation $[a,b]$ denotes the $\sim_{\mathbb{Q}}$-equivalence class of the
prefraction $(a,b)$, where $b\ne 0_{\mathbb{Z}}$.
\end{definition}
\end{definitionbox}

\begin{dependencies}
\begin{itemize}
  \item \hyperref[def:rationals]{Rational Numbers}
  \item \hyperref[def:fraction-equivalence]{Fraction Equivalence}
\end{itemize}
\end{dependencies}

\begin{propositionbox}{Proposition}
\begin{proposition}[Equality Criterion for Rational Classes]
\label{prop:equality-criterion-for-rational-classes}
For prefractions $(a,b),(c,d)\in\operatorname{Pre}\mathbb{Q}$,
\[
[a,b]=[c,d]
\quad\Longleftrightarrow\quad
a\cdot d=b\cdot c.
\]
\end{proposition}
\end{propositionbox}
