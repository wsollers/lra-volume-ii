\subsection*{No Rational Square Root of Two}

\begin{theorembox}{Theorem}
\begin{theorem}[No Rational Square Root of Two]
\label{thm:no-rational-square-root-of-two}
There is no rational number $q\in\mathbb{Q}$ such that
\[
q\cdot q=2_{\mathbb{Q}}.
\]
\end{theorem}
\end{theorembox}

\subsection*{The Square-Two Gap}

\begin{definitionbox}{Definition}
\begin{definition}[The Gap Set in $\mathbb{Q}$]
\label{def:gap-set-q}
Define
\[
S:=\{q\in\mathbb{Q}:0_{\mathbb{Q}}<q\ \text{and}\ q\cdot q<2_{\mathbb{Q}}\}.
\]
\end{definition}

\subsection*{Cauchy Sequences in $\mathbb{Q}$}

\begin{definitionbox}{Definition}
\begin{definition}[Cauchy Sequence in $\mathbb{Q}$]
\label{def:cauchy-sequence-in-q}
A sequence $(x_n)$ in $\mathbb{Q}$ is Cauchy if
\[
\forall\varepsilon\in\mathbb{Q}\,
\left(
0_{\mathbb{Q}}<\varepsilon
\Longrightarrow
\exists N\in\mathbb{N}\,
\forall m,n\ge N\,
\left|x_m-x_n\right|<\varepsilon
\right).
\]
\end{definition}
\end{definitionbox}
