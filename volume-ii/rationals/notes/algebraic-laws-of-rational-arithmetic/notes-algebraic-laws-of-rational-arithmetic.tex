\section{Algebraic Laws of Rational Arithmetic}
\label{sec:algebraic-laws-rational-arithmetic}

The previous section proves that the operations on rational classes are well-defined.  The next step is to prove that these operations obey the ordinary algebraic laws expected of fractions.

\begin{remark*}[Existing proof inventory]
The detailed algebraic law proofs are currently packaged in
\hyperref[sec:field-structure-q]{Field Structure}.  This section is the intended home for the representative-level proofs of the individual laws before they are bundled into the theorem that $\mathbb{Q}$ is a field.
\end{remark*}

\begin{tcolorbox}[colback=thmbox, colframe=thmborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Theorem (Rational Addition Is Commutative)}},
  fonttitle=\small\bfseries]
\begin{theorem}[Rational Addition Is Commutative]
\label{thm:rational-addition-commutative}
For all $x,y\in\mathbb{Q}$,
\[
x+y=y+x.
\]
\end{theorem}
\end{tcolorbox}

\begin{remark*}[Status]
Missing as an individual representative-level proof. It is included implicitly in \hyperref[thm:q-is-integral-domain]{Theorem~\ref*{thm:q-is-integral-domain}}.
\end{remark*}

\begin{tcolorbox}[colback=thmbox, colframe=thmborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Theorem (Rational Addition Is Associative)}},
  fonttitle=\small\bfseries]
\begin{theorem}[Rational Addition Is Associative]
\label{thm:rational-addition-associative}
For all $x,y,z\in\mathbb{Q}$,
\[
(x+y)+z=x+(y+z).
\]
\end{theorem}
\end{tcolorbox}

\begin{remark*}[Status]
Missing as an individual representative-level proof. It is included implicitly in \hyperref[thm:q-is-integral-domain]{Theorem~\ref*{thm:q-is-integral-domain}}.
\end{remark*}

\begin{tcolorbox}[colback=thmbox, colframe=thmborder, arc=2pt,
  left=6pt, right=6pt, top=4pt, bottom=4pt,
  title={\small\textbf{Theorem (Rational Additive Identity)}},
  fonttitle=\small\bfseries]
\begin{theorem}[Rational Additive Identity]
\label{thm:rational-additive-identity}
For every $x\in\mathbb{Q}$,
\[
x+0=x=0+x.
\]
\end{theorem}
\end{tcolorbox}

\begin{remark*}[Status]
Missing as an individual representative-level proof. Expected representative: $0=[(0,1)]$.
\end{remark*}

\end{theorem}
\end{tcolorbox}
