\subsection*{The Integer Embedding}

\begin{definitionbox}{Definition}
\begin{definition}[Embedding of $\mathbb{Z}$ into $\mathbb{Q}$]
\label{def:rational-chapter-embedding-z-into-q}
Define
\[
\iota:\mathbb{Z}\to\mathbb{Q},
\qquad
\iota(n):=[n,1_{\mathbb{Z}}].
\]
\end{definition}
\end{definitionbox}

\begin{theorem}[Embedding $\mathbb{Z}$ into $\mathbb{Q}$ Is Injective]

\subsection*{Preservation of Algebra and Order}

\begin{theorembox}{Theorem}
\begin{theorem}[Embedding Preserves Addition]
\label{thm:embedding-z-into-q-preserves-addition}
For all $m,n\in\mathbb{Z}$,
\[
\iota(m+n)=\iota(m)+\iota(n).
\]
\end{theorem}
\end{theorembox}

\subsection*{Field of Fractions}

\begin{theorembox}{Theorem}
\begin{theorem}[$\mathbb{Q}$ Is the Field of Fractions of $\mathbb{Z}$]
\label{thm:q-is-field-of-fractions-of-z}
For every $x\in\mathbb{Q}$, there exist $m,n\in\mathbb{Z}$ with
$n\ne0_{\mathbb{Z}}$ such that
\[
x=\iota(m)\cdot \iota(n)^{-1}.
\]
\end{theorem}
\end{theorembox}
