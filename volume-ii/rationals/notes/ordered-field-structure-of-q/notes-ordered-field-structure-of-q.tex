\subsection*{Ordered Field Instantiation}

\begin{theorembox}{Theorem}
\begin{theorem}[$\mathbb{Q}$ Is an Ordered Field]
\label{thm:q-is-an-ordered-field}
The structure
\[
(\mathbb{Q},+,\cdot,0_{\mathbb{Q}},1_{\mathbb{Q}},\le)
\]
is an ordered field.
\end{theorem}
\end{theorembox}

\subsection*{Standard Ordered-Field Consequences}

\begin{corollarybox}{Corollary}
\begin{corollary}[Squares Are Nonnegative on $\mathbb{Q}$]
\label{cor:squares-nonnegative-on-q}
For every $x\in\mathbb{Q}$,
\[
0_{\mathbb{Q}}\le x^2.
\]
\end{corollary}
\end{corollarybox}

\begin{corollarybox}{Corollary}
\begin{corollary}[One Is Positive in $\mathbb{Q}$]
\label{cor:one-positive-on-q}
In $\mathbb{Q}$,
\[
0_{\mathbb{Q}}<1_{\mathbb{Q}}.
\]
\end{corollary}
\end{corollarybox}

\begin{corollarybox}{Corollary}
\begin{corollary}[Natural Numbers Are Positive in $\mathbb{Q}$]
\label{cor:natural-numbers-positive-on-q}
Every natural number embedded in $\mathbb{Q}$ is positive.
\end{corollary}
\end{corollarybox}

\begin{corollarybox}{Corollary}
\begin{corollary}[Positive Rationals Have Positive Inverses]
\label{cor:positive-rationals-have-positive-inverses}
For every $x\in\mathbb{Q}$,
\[
0_{\mathbb{Q}}<x
\Longrightarrow
0_{\mathbb{Q}}<x^{-1}.
\]
\end{corollary}
\end{corollarybox}

\begin{corollarybox}{Corollary}
\begin{corollary}[Fraction Comparison in $\mathbb{Q}$]
\label{cor:fraction-comparison-on-q}
For rational classes with positive denominator product,
\[
[a,b]<[c,d]
\quad\Longleftrightarrow\quad
a\cdot d<c\cdot b.
\]
\end{corollary}
\end{corollarybox}
