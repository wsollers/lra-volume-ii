\subsection*{Integer Powers on $\mathbb{Q}$}

\begin{definitionbox}{Definition}
\begin{definition}[Integer Power on $\mathbb{Q}$]
\label{def:integer-power-on-q}
Let $x\in\mathbb{Q}$. For $n\in\mathbb{W}$, define $x^n$ by the usual
recursive power operation. If $x\ne 0_{\mathbb{Q}}$ and $n\in\mathbb{Z}$ is
negative, define
\[
x^n:=(x^{-1})^{-n}.
\]
\end{definition}
\end{definitionbox}
