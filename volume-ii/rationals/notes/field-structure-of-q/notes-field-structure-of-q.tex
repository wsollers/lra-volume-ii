\section{Field Structure}
\label{sec:field-structure-q}

Once addition and multiplication are shown to be well-defined on
equivalence classes of fraction pairs, the rational numbers become much
more than a construction. They become an algebraic number system in
their own right.

\begin{definition}[Integral domain]
\label{def:integral-domain}
A system
\[
(F,+,\cdot,0,1)
\]
is called an \emph{integral domain} if:
\begin{enumerate}
  \item $(F,+,0)$ is a commutative group;
  \item multiplication on $F$ is associative and commutative;
  \item $1\neq 0$ and
  \[
  \forall a\in F,\qquad 1\cdot a=a=a\cdot 1;
  \]
  \item multiplication distributes over addition;
  \item $F$ has no zero divisors, that is,
  \[
  \forall a,b\in F,\qquad ab=0 \Longrightarrow a=0 \text{ or } b=0.
  \]
\end{enumerate}
\end{definition}

\begin{definition}[Field]
\label{def:field}
A system
\[
(F,+,\cdot,0,1)
\]
is called a \emph{field} if it is an integral domain and satisfies:
\[
\forall a,b\in F,\qquad b\neq 0 \Longrightarrow \exists x\in F\;(b\cdot x=a).
\]
Equivalently, every nonzero element of $F$ has a multiplicative inverse.
\end{definition}

\begin{definition}[Subfield]
\label{def:subfield}
Let
\[
(E,+,\cdot,0,1)
\qquad\text{and}\qquad
(F,+,\cdot,0,1)
\]
be fields with $E\subseteq F$. We say that $E$ is a \emph{subfield} of $F$
if the operations and distinguished elements on $E$ are the restrictions of
those on $F$.
\end{definition}

\begin{definition}[Nonnegative integer exponents in a field]
\label{def:nonneg-integer-exponents-field}
Let $F$ be a field, let $x\in F$, and let $n\in\mathbb{N}$.  Define
\[
x^0 := 1
\qquad\text{and}\qquad
x^{n+1} := x^n\cdot x.
\]
In particular $x^1 = x$ and $x^2 = x\cdot x$.
\end{definition}

\begin{remark}
This is the same recursive definition used for natural-number exponentiation
in $\mathbb{N}$, now stated inside an arbitrary field.  No inversion is
required and the definition is valid for all $x\in F$, including $x=0$.
\end{remark}

\begin{definition}[Negative integer exponents in a field]
Let $F$ be a field, let $x\in F$ with $x\neq 0$, and let $m\in\mathbb{N}$.
Define
\[
x^{-m}:=(x^{-1})^m.
\]
\end{definition}

\begin{remark}
This theorem packages the algebraic facts that follow from the quotient

\end{remark}

\begin{theorem}[$\mathbb{Q}$ Is a Field]
\label{thm:q-is-field}
With the operations of
\hyperref[def:rational-operations]{Definition~\ref*{def:rational-operations}},
the system
\[
(\mathbb{Q},+,\cdot,0,1)
\]
is a field.
\end{theorem}

\begin{remark*}[Proof]
\hyperref[prf:q-is-field]{Go to proof.}
\end{remark*}

\begin{remark}[What structure is gained]
To say that $\mathbb{Q}$ is a field means, in particular, that:
\begin{itemize}
  \item addition and multiplication are closed on $\mathbb{Q}$;
  \item both operations are associative and commutative;
  \item $0$ and $1$ serve as additive and multiplicative identities;
  \item every rational has an additive inverse;
  \item every nonzero rational has a multiplicative inverse;
  \item multiplication distributes over addition.
\end{itemize}
\end{remark}

\begin{remark}[Why this matters]
This is the point where division becomes internal to the number system.
The integers allowed subtraction, but not division by arbitrary nonzero
elements. The rationals repair exactly that failure.
\end{remark}

\begin{theorem}[Negative Exponents Agree with Inverse Powers]
\label{thm:negative-exponents-agree-with-inverse-power}
Let $F$ be a field, let $x\in F$ with $x\neq 0$, and let $m,n\in\mathbb{Z}$.
Then
\[
x^{-m}=(x^{-1})^m=(x^m)^{-1}.
\]
Moreover, the usual laws of exponents continue to hold for integer exponents.
\end{theorem}

\begin{theorem}[Power of a Product in a Field]
\label{thm:power-of-product-in-field}
Let $(F,+,\cdot,0,1)$ be a field. Let $q,r\in F\setminus\{0\}$ and let
$a\in\mathbb{Z}$. Then
\[
(qr)^a=q^a r^a.
\]
\end{theorem}

\begin{remark}[Elementary consequences of field axioms]
\label{rem:field-identity-consequences}
The following identities hold in any field.  They are consequences of
the field axioms alone and require no order structure.  They are stated
here because they are invoked silently throughout the rational algebra
proofs; naming them gives those proofs clean citation targets.
\end{remark}

\begin{theorem}[Zero product]
\label{thm:zero-product}
Let $F$ be a field and let $a\in F$.  Then
\[
a\cdot 0 = 0.
\]
\end{theorem}

\begin{theorem}[Double negation]
\label{thm:double-negation}
Let $F$ be a field and let $a\in F$.  Then
\[

\begin{theorem}[Negation of a product]
\label{thm:negation-of-product}
Let $F$ be a field and let $a,b\in F$.  Then
\[
-(a\cdot b) = (-a)\cdot b = a\cdot(-b).

\begin{theorem}[Additive cancellation]
\label{thm:additive-cancellation}
Let $F$ be a field and let $a,b,c\in F$.  If
\[
a + c = b + c,
\]
then $a = b$.
\end{theorem}

\begin{theorem}[Multiplicative cancellation]
\label{thm:multiplicative-cancellation}
Let $F$ be a field and let $a,b,c\in F$ with $c\neq 0$.  If
\[
a\cdot c = b\cdot c,

\begin{theorem}[Uniqueness of multiplicative inverse]
\label{thm:uniqueness-of-multiplicative-inverse}
Let $F$ be a field and let $a\in F$ with $a\neq 0$.  If $b\in F$

\begin{remark}[Absolute value]
\label{rem:absolute-value-forward-ref}
An ordered field carries a natural notion of absolute value: for
$x\in F$ define $|x|=x$ if $x\geq 0$ and $|x|=-x$ if $x<0$.
The full definition and its standard properties --- non-negativity,
multiplicativity, and the triangle inequality --- are developed in
the modulus chapter of Volume~III once the order machinery is fully
in place.
All uses of $|{\cdot}|$ in the sequence and Cauchy sections below
anticipate that treatment; the reader should treat them as shorthand
for the piecewise definition above.
\end{remark}

\begin{remark}[What comes next]
Field structure alone does not yet explain inequalities, positivity, or
approximation. Those belong to the order story, which is why the next
step is to endow $\mathbb{Q}$ with order axioms compatible with the
field operations.
\end{remark}
